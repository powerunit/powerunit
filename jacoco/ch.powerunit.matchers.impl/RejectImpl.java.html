package ch.powerunit.matchers.impl;

import org.hamcrest.Matcher;

import ch.powerunit.matchers.lang.MatcherAssertion.Reject;

/**
 * @author borettim
 * @since 0.3.0
 */
public class RejectImpl implements Reject {
	private final Object value;
	private final Matcher<String> messageValidation;
	
	
	/**
	 * @param value
	 * @param messageValidation
	 */
	public RejectImpl(Object value, Matcher<String> messageValidation) {
		this.value = value;
		this.messageValidation = messageValidation;
	}
	/**
	 * @return the value
	 */
	public Object getValue() {
		return value;
	}
	/**
	 * @return the messageValidation
	 */
	public Matcher<String> getMessageValidation() {
		return messageValidation;
	}
	
	
}
