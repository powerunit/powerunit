20  package ch.powerunit;
21  
22  import java.util.Calendar;
23  import java.util.Date;
24  import java.util.Map;
25  import java.util.Objects;
26  import java.util.function.Function;
27  
28  /**
29   * This is the converter support.
30   * 
31   * @author borettim
32   * @since 0.4.0
33   */
34  interface ConverterMethod {
35  
36  	/**
37  	 * Return a converter that return null if the receiver is null or else
38  	 * return the result of the received converter.
39  	 * 
40  	 * @param converter
41  	 *            the converter to be used if the received is not null.
42  	 * @return the converter that support the null value.
43  	 * @param <T>
44  	 *            input type of the converter.
45  	 * @param <R>
46  	 *            output type of the converter.
47  	 * @since 0.4.0
48  	 */
49  	default <T, R> Function<T, R> nullToNullConverter(Function<T, R> converter) {
50  		Objects.requireNonNull(converter, "converter can't be null");
51  		return (p) -> {
52  			if (p == null) {
53  				return null;
54  			}
55  			return converter.apply(p);
56  		};
57  	}
58  
59  	/**
60  	 * Return a function that convert {@link Date} to {@link Calendar}.
61  	 * <p>
62  	 * This function will return {@code null} when the received date is
63  	 * {@code null}.
64  	 * 
65  	 * @return a {@link Function} to converter {@link Date} to {@link Calendar}.
66  	 * @since 0.4.0
67  	 */
68  	default Function<Date, Calendar> dateToCalendar() {
69  		return nullToNullConverter((Date p) -> {
70  			Calendar c = Calendar.getInstance();
71  			c.setTime(p);
72  			return c;
73  		});
74  	}
75  
76  	/**
77  	 * Return a function that convert {@link Calendar} to {@link Date}.
78  	 * <p>
79  	 * This function will return {@code null} when the received Calendar is
80  	 * {@code null}.
81  	 * 
82  	 * @return a {@link Function} to converter {@link Calendar} to {@link Date}.
83  	 * @since 0.4.0
84  	 */
85  	default Function<Calendar, Date> calendarToDate() {
86  		return nullToNullConverter((p) -> p.getTime());
87  	}
88  
89  	/**
90  	 * Return a function that get a value from a {@link Map}, as converter.
91  	 * 
92  	 * @param key
93  	 *            the key to be used to retrieve the value from the {@link Map}
94  	 *            ; Can't be null.
95  	 * @return a {@link Function} to get a value from a {@link Map}.
96  	 * @param <K>
97  	 *            the type of the Key.
98  	 * @param <V>
99  	 *            the type of the Value.
100 	 * @since 0.4.0
101 	 */
102 	default <K, V> Function<Map<K, V>, V> mapToValue(K key) {
103 		Objects.requireNonNull(key, "key can't be null");
104 		return nullToNullConverter((Map<K, V> p) -> p.getOrDefault(key, null));
105 	}
106 
107 	/**
108 	 * Functional Interface for exception method conversion.
109 	 * 
110 	 * @author borettim
111 	 * @since 0.4.0
112 	 */
113 	@FunctionalInterface
114 	interface NoArgumentWithThrowableMethod {
115 		void execute() throws Throwable;
116 	}
117 
118 	/**
119 	 * Converter a piece of code to statement.
120 	 * 
121 	 * @param method
122 	 *            the no arg no return piece of code to be tested.
123 	 * @return the statement
124 	 * @since 0.4.0
125 	 */
126 	default Statement<?, Throwable> asStatement(
127 			NoArgumentWithThrowableMethod method) {
128 		return (p) -> {
129 			method.execute();
130 		};
131 	}
132 }
