20  package ch.powerunit.pattern.impl;
21  
22  import java.util.List;
23  import java.util.function.Supplier;
24  import java.util.regex.Pattern;
25  import java.util.stream.Stream;
26  import java.util.stream.Stream.Builder;
27  
28  import org.hamcrest.Matcher;
29  
30  import ch.powerunit.Parameter;
31  import ch.powerunit.Parameters;
32  import ch.powerunit.Rule;
33  import ch.powerunit.Test;
34  import ch.powerunit.TestDelegate;
35  import ch.powerunit.TestDelegator;
36  import ch.powerunit.TestRule;
37  import ch.powerunit.TestSuite;
38  import ch.powerunit.pattern.PatternTester;
39  
40  /**
41   * @author borettim
42   *
43   */
44  @TestDelegator
45  public class PatternTesterImpl implements TestSuite {
46  	@Parameters
47  	public static Stream<Object[]> getParameters(PatternTester input) {
48  		Builder<Object[]> builder = Stream.builder();
49  		for (int i = 0; i < input.getInputs().size(); i++) {
50  			builder.add(new Object[] { input.getUnderTest(),
51  					input.getInputs().get(i), input.getExpectedResult().get(i),
52  					input.getHavingGroup().get(i),
53  					input.getExpectedGroup().get(i),
54  					input.getUnderTest().pattern() });
55  		}
56  		return builder.build();
57  	}
58  
59  	@Parameter(0)
60  	public Pattern underTest;
61  
62  	@Parameter(1)
63  	public String receiving;
64  
65  	@Parameter(2)
66  	public boolean expectedResult;
67  
68  	@Parameter(3)
69  	public List<Integer> groups;
70  
71  	@Parameter(4)
72  	public List<Matcher<String>> expectings;
73  
74  	@Parameter(5)
75  	public String patternAsString;
76  
77  	private java.util.regex.Matcher result;
78  
79  	private boolean matches;
80  
81  	@Rule
82  	public final TestRule rules = before(this::prepare);
83  
84  	private void prepare() {
85  		result = underTest.matcher(receiving);
86  		matches = result.matches();
87  	}
88  
89  	@TestDelegate
90  	public final Supplier<MatcherTester> matcherTester = () -> new MatcherTester(
91  			result, receiving, groups, expectings);
92  
93  	@Test(name = "Validate that Pattern `%6$s` matches `%2$s` is `%3$s`")
94  	public void testMatches() {
95  		assertThat(matches).is(expectedResult);
96  	}
97  }
