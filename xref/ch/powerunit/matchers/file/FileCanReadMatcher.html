20  package ch.powerunit.matchers.file;
21  
22  import java.io.File;
23  
24  import org.hamcrest.FeatureMatcher;
25  import org.hamcrest.Matcher;
26  
27  /**
28   * The Mather on can read File
29   *
30   * @author borettim
31   *
32   */
33  public class FileCanReadMatcher extends FeatureMatcher<File, Boolean> {
34  
35      public FileCanReadMatcher(Matcher<? super Boolean> subMatcher) {
36          super(subMatcher, "canRead", "canRead");
37      }
38  
39      @Override
40      protected Boolean featureValueOf(File actual) {
41          return actual.canRead();
42      }
43  
44  }
