20  package ch.powerunit.matchers.lang;
21  
22  import org.hamcrest.Matcher;
23  
24  /**
25   * Use to define the expectation on one single instance of a matcher.
26   * 
27   * @author borettim
28   * @since 0.3.0
29   */
30  public interface MatcherAssertion<T extends Matcher<?>> {
31  	/**
32  	 * Indicate that this matcher instance must accept the null value.
33  	 * 
34  	 * @return continuation of the DSL.
35  	 */
36  	MatcherAssertion<T> nullAccepted();
37  
38  	/**
39  	 * Indicate that this matcher instance must reject the null value with a
40  	 * message accepted by a matcher.
41  	 * 
42  	 * @param rejectedMessage
43  	 *            the matcher on the message
44  	 * @return continuation of the DSL
45  	 */
46  	MatcherAssertion<T> nullRejected(Matcher<String> rejectedMessage);
47  
48  	/**
49  	 * Indicate that this matcher instance must reject the null value with a
50  	 * message.
51  	 * 
52  	 * @param rejectedMessage
53  	 *            the expected message
54  	 * @return continuation of the DSL
55  	 */
56  	MatcherAssertion<T> nullRejected(String rejectedMessage);
57  
58  	/**
59  	 * Define a list of value accepted by this matcher.
60  	 * 
61  	 * @param values
62  	 *            the values accepted by this matcher
63  	 * @return continuation of the DSL
64  	 */
65  	MatcherAssertion<T> accepting(Object... values);
66  
67  	/**
68  	 * Define a list of value rejected by this matcher.
69  	 * 
70  	 * @param reject
71  	 *            the values (and expectation on rejecting message) rejected by
72  	 *            this this matcher.
73  	 * @return continuation of the DSL
74  	 * @see ch.powerunit.matchers.MatcherTester#value(Object)
75  	 */
76  	MatcherAssertion<T> rejecting(Reject... reject);
77  
78  	/**
79  	 * Definition of a rejected value.
80  	 * 
81  	 * @author borettim
82  	 * @since 0.3.0
83  	 */
84  	interface Reject {
85  
86  	}
87  }
