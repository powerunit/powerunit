20  package ch.powerunit.matchers;
21  
22  import org.hamcrest.Matcher;
23  import org.hamcrest.Matchers;
24  
25  import ch.powerunit.TestInterface;
26  import ch.powerunit.matchers.impl.MatcherAssertionImpl;
27  import ch.powerunit.matchers.impl.MatcherTesterImpl;
28  import ch.powerunit.matchers.impl.RejectImpl;
29  import ch.powerunit.matchers.lang.MatcherAssertion;
30  import ch.powerunit.matchers.lang.MatcherTesterDSL1;
31  import ch.powerunit.matchers.lang.MatcherTesterDSL2;
32  import ch.powerunit.matchers.lang.MatcherTesterDSL3;
33  import ch.powerunit.matchers.lang.MatcherAssertion.Reject;
34  
35  /**
36   * This class provide the support for tester of Matcher.
37   * 
38   * @author borettim
39   * @since 0.3.0
40   * @see #of(Class) To start the DSL.
41   */
42  @TestInterface(MatcherTesterImpl.class)
43  public final class MatcherTester<T extends Matcher<?>> {
44  	/**
45  	 * Start the creation of a tester definition for a matcher.
46  	 * <p>
47  	 * For instance :
48  	 * 
49  	 * <pre>
50  	 * &#064;TestDelegate
51  	 * public final MatcherTester&lt;TestMatcher&gt; tester = MatcherTester.of(
52  	 * 		TestMatcher.class).with(
53  	 * 		matcher(new TestMatcher(null)).describedAs(&quot;is null&quot;).nullAccepted()
54  	 * 				.rejecting(value(&quot;x&quot;).withMessage(&quot; was \&quot;x\&quot;&quot;)),
55  	 * 		matcher(new TestMatcher(&quot;x&quot;)).describedAs(&quot;is \&quot;x\&quot;&quot;)
56  	 * 				.nullRejected(&quot; was null&quot;).accepting(&quot;x&quot;)
57  	 * 				.rejecting(value(&quot;y&quot;).withMessage(&quot; was \&quot;y\&quot;&quot;)),
58  	 * 		matcher(new TestMatcher(&quot;y&quot;)).describedAs(&quot;is \&quot;y\&quot;&quot;)
59  	 * 				.nullRejected(&quot; was null&quot;).accepting(&quot;y&quot;)
60  	 * 				.rejecting(value(&quot;x&quot;).withMessage(&quot; was \&quot;x\&quot;&quot;)));
61  	 * </pre>
62  	 * 
63  	 * @param matcherClass
64  	 *            the class of the matcher
65  	 * @return the next step of the DSL
66  	 * @see ch.powerunit.matchers.lang.MatcherTesterDSL1#with(MatcherAssertion...)
67  	 */
68  	public static <T extends Matcher<?>> MatcherTesterDSL1<T> of(
69  			Class<T> matcherClass) {
70  		return new MatcherTesterDSL1<T>() {
71  
72  			@Override
73  			public MatcherTester<T> with(MatcherAssertion<T>... assertions) {
74  				return new MatcherTester<T>(matcherClass, assertions);
75  			}
76  
77  			@SuppressWarnings("unchecked")
78  			@Override
79  			public MatcherTester<T> with(MatcherAssertion<T> first) {
80  				return with(new MatcherAssertion[] { first });
81  			}
82  
83  			@SuppressWarnings("unchecked")
84  			@Override
85  			public MatcherTester<T> with(MatcherAssertion<T> first,
86  					MatcherAssertion<T> second) {
87  				return with(new MatcherAssertion[] { first, second });
88  			}
89  
90  			@SuppressWarnings("unchecked")
91  			@Override
92  			public MatcherTester<T> with(MatcherAssertion<T> first,
93  					MatcherAssertion<T> second, MatcherAssertion<T> third) {
94  				return with(new MatcherAssertion[] { first, second, third });
95  			}
96  
97  			@SuppressWarnings("unchecked")
98  			@Override
99  			public MatcherTester<T> with(MatcherAssertion<T> first,
100 					MatcherAssertion<T> second, MatcherAssertion<T> third,
101 					MatcherAssertion<T> fourth) {
102 				return with(new MatcherAssertion[] { first, second, third,
103 						fourth });
104 			}
105 
106 			@SuppressWarnings("unchecked")
107 			@Override
108 			public MatcherTester<T> with(MatcherAssertion<T> first,
109 					MatcherAssertion<T> second, MatcherAssertion<T> third,
110 					MatcherAssertion<T> fourth, MatcherAssertion<T> fifth) {
111 				return with(new MatcherAssertion[] { first, second, third,
112 						fourth, fifth });
113 			}
114 
115 			@SuppressWarnings("unchecked")
116 			@Override
117 			public MatcherTester<T> with(MatcherAssertion<T> first,
118 					MatcherAssertion<T> second, MatcherAssertion<T> third,
119 					MatcherAssertion<T> fourth, MatcherAssertion<T> fifth,
120 					MatcherAssertion<T> sixth) {
121 				return with(new MatcherAssertion[] { first, second, third,
122 						fourth, fifth, sixth });
123 			}
124 		};
125 	}
126 
127 	/**
128 	 * Use this method to start the definition of one entry to be passed into
129 	 * the <code>with</code> created by {@link #of(Class)}.
130 	 * 
131 	 * @param matcher
132 	 *            The instance of the matcher to be tested.
133 	 * @return the next step of the DSL
134 	 * @see ch.powerunit.matchers.lang.MatcherTesterDSL2
135 	 */
136 	public static <T extends Matcher<?>> MatcherTesterDSL2<T> matcher(T matcher) {
137 		return new MatcherTesterDSL2<T>() {
138 
139 			@Override
140 			public MatcherAssertion<T> describedAs(Matcher<String> expected) {
141 				return new MatcherAssertionImpl<T>(matcher, expected);
142 			}
143 
144 			@Override
145 			public MatcherAssertion<T> describedAs(String expected) {
146 				return describedAs(Matchers.equalTo(expected));
147 			}
148 
149 		};
150 	}
151 
152 	/**
153 	 * Use this method inside the
154 	 * {@link ch.powerunit.matchers.lang.MatcherAssertion#rejecting(Reject...)}
155 	 * to start the definition of a rejecting object.
156 	 * 
157 	 * @param value
158 	 *            The rejected value
159 	 * @return the next step of the DSL
160 	 * @see ch.powerunit.matchers.lang.MatcherTesterDSL3
161 	 */
162 	public static MatcherTesterDSL3 value(Object value) {
163 		return new MatcherTesterDSL3() {
164 
165 			@Override
166 			public Reject withMessage(Matcher<String> message) {
167 				return new RejectImpl(value, message);
168 			}
169 
170 			@Override
171 			public Reject withMessage(String message) {
172 				return withMessage(Matchers.equalTo(message));
173 			}
174 		};
175 	}
176 
177 	private final Class<T> matcherClass;
178 
179 	private final MatcherAssertion<T> assertions[];
180 
181 	/**
182 	 * @param matcherClass
183 	 * @param assertions
184 	 */
185 	private MatcherTester(Class<T> matcherClass,
186 			MatcherAssertion<T>[] assertions) {
187 		this.matcherClass = matcherClass;
188 		this.assertions = assertions;
189 	}
190 
191 	/**
192 	 * <b>Used internally</b>.
193 	 * 
194 	 * @return the matcherClass
195 	 */
196 	public Class<T> getMatcherClass() {
197 		return matcherClass;
198 	}
199 
200 	/**
201 	 * <b>Used internally</b>.
202 	 * 
203 	 * @return the assertions
204 	 */
205 	public MatcherAssertion<T>[] getAssertions() {
206 		return assertions;
207 	}
208 
209 }
