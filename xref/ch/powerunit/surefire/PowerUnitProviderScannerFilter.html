20  package ch.powerunit.surefire;
21  
22  import java.lang.reflect.Method;
23  import java.util.Collections;
24  import java.util.HashSet;
25  import java.util.Properties;
26  import java.util.Set;
27  
28  import org.apache.maven.surefire.booter.ProviderParameterNames;
29  import org.apache.maven.surefire.util.ScannerFilter;
30  
31  import ch.powerunit.Test;
32  
33  /**
34   * @author borettim
35   *
36   */
37  public class PowerUnitProviderScannerFilter implements ScannerFilter {
38  
39  	private final Set<String> sgroups;
40  
41  	private final Set<String> sexcludedGroups;
42  
43  	public PowerUnitProviderScannerFilter(Properties parameters) {
44  		String groups = parameters.getProperty(
45  				ProviderParameterNames.TESTNG_GROUPS_PROP, "");
46  		String excludedGroups = parameters.getProperty(
47  				ProviderParameterNames.TESTNG_EXCLUDEDGROUPS_PROP, "");
48  		Set<String> sgroups = new HashSet<String>();
49  		Set<String> sexcludedGroups = new HashSet<String>();
50  		for (String g : groups.split(",")) {
51  			sgroups.add(g);
52  		}
53  		for (String g : excludedGroups.split(",")) {
54  			sexcludedGroups.add(g);
55  		}
56  		this.sgroups = Collections.unmodifiableSet(sgroups);
57  		this.sexcludedGroups = Collections.unmodifiableSet(sexcludedGroups);
58  	}
59  
60  	@Override
61  	public boolean accept(@SuppressWarnings("rawtypes") Class testClass) {
62  		for (Method m : testClass.getDeclaredMethods()) {
63  			if (m.isAnnotationPresent(Test.class)) {
64  				return true;
65  			}
66  		}
67  		return false;
68  	}
69  }
