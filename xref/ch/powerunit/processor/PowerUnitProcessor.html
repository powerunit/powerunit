51  @SupportedAnnotationTypes({ "ch.powerunit.Test", "ch.powerunit.Rule",
52          "ch.powerunit.Parameters", "ch.powerunit.Parameter",
53          "ch.powerunit.Ignore", "ch.powerunit.Categories",
54          "ch.powerunit.TestDelegate" })
55  @SupportedSourceVersion(SourceVersion.RELEASE_8)
56  public class PowerUnitProcessor extends AbstractProcessor implements
57          ParametersProcessorValidator, ParameterProcessorValidator,
58          RuleProcessorValidator, TestProcessorValidator,
59          IgnoreProcessorValidator, TestDelegateProcessorValidator {
60  
61      private boolean categoryDone = false;
62  
63      @Override
64      public boolean process(Set<? extends TypeElement> annotations,
65              RoundEnvironment roundEnv) {
66          if (!roundEnv.processingOver()) {
67              testAnnotationValidation(roundEnv);
68              ruleAnnotationValidation(processingEnv, roundEnv);
69              parametersValidation(processingEnv, roundEnv);
70              parameterAnnotationValidation(processingEnv, roundEnv);
71              ignoreAnnotationValidation(roundEnv);
72              testDelegateAnnotationValidation(processingEnv, roundEnv);
73              try {
74                  if (!categoryDone) {
75                      generateCategorySuite(roundEnv);
76                  }
77              } catch (IOException e) {
78                  error("Unable to generate test suite file, because of "
79                          + e.getMessage());
80              }
81          }
82  
83          return true;
84      }
85  
86      private void generateCategorySuite(RoundEnvironment roundEnv)
87              throws IOException {
88          categoryDone = true;
89          Set<? extends Element> elements = roundEnv
90                  .getElementsAnnotatedWith(Categories.class);
91          Map<String, List<TypeMirror>> categories = new HashMap<>();
92          for (Element element : elements) {
93              if (element.getKind() == ElementKind.CLASS) {
94                  TypeMirror current = element.asType();
95                  Categories ca = element.getAnnotation(Categories.class);
96                  if (ca != null) {
97                      String cat[] = ca.value();
98                      for (String c : cat) {
99                          List<TypeMirror> lst = categories.get(c);
100                         if (lst == null) {
101                             lst = new ArrayList<>();
102                         }
103                         lst.add(current);
104                         categories.put(c, lst);
105                     }
106                 }
107             }
108         }
109         JavaFileObject suite = processingEnv.getFiler()
110                 .createSourceFile("ch.powerunit.suite.Suites",
111                         elements.toArray(new Element[] {}));
112         try (PrintWriter w = new PrintWriter(suite.openWriter());) {
113             w.println("package ch.powerunit.suite;");
114             w.println();
115             w.println("@javax.annotation.Generated(\"Test suites for powerunit\")");
116             w.println("public final class Suites {");
117             w.println();
118             w.println();
119             w.println("  private static final java.util.Map<String,Class<?>[]> suites;");
120             w.println();
121             w.println("  static {");
122             w.println("    java.util.Map<String,Class<?>[]> tmp = new java.util.HashMap<>();");
123             for (Map.Entry<String, List<TypeMirror>> cat : categories
124                     .entrySet()) {
125                 w.println("    tmp.put(\"" + cat.getKey()
126                         + "\",new Class<?>[]{");
127                 int counter = 0;
128                 for (TypeMirror tm : cat.getValue()) {
129                     if (counter > 0) {
130                         w.println(",");
131                     }
132                     w.print("      "
133                             + ((DeclaredType) tm).asElement().toString()
134                             + ".class");
135                     counter++;
136                 }
137                 w.println();
138                 w.println("    });");
139             }
140             w.println("    suites=java.util.Collections.unmodifiableMap(tmp);");
141             w.println("  }");
142             w.println();
143             w.println("  public static java.util.Map<String,Class<?>[]> getSuites() {");
144             w.println("    return suites;");
145             w.println("  }");
146             w.println();
147             w.println();
148             w.println();
149             w.println("  public static void main(String argv[]) {");
150             w.println("    if (argv.length==1) {");
151             w.println("       suites.keySet().stream().forEach((c)->{main(new String[]{argv[0],c});});");
152             w.println("    return;");
153             w.println("    }");
154             w.println("    String name = argv[1];");
155             w.println("    StringBuilder sb = new StringBuilder();");
156             w.println("    for(Class<?> c:getSuites().get(name)) {");
157             w.println("      sb.append(c.getName()).append(',');");
158             w.println("    }");
159             w.println("    ch.powerunit.PowerUnitMainRunner.main(new String[]{argv[0],sb.toString()});");
160             w.println("  }");
161             w.println();
162             w.println("}");
163         }
164     }
165 
166     @Override
167     public String elementAsString(Element ee) {
168         return ee.toString() + " of " + ee.getEnclosingElement().toString();
169     }
170 
171     @Override
172     public void error(String msg) {
173         processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, msg);
174     }
175 
176     @Override
177     public void warn(String msg) {
178         processingEnv.getMessager().printMessage(
179                 Diagnostic.Kind.MANDATORY_WARNING, msg);
180     }
