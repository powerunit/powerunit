20  package ch.powerunit;
21  
22  import java.util.function.Function;
23  
24  /**
25   * This is an intermediate and optional step on assertion to cast an object into
26   * a sub-object.
27   * 
28   * @author borettim
29   * @since 0.3.0
30   * @param <T>
31   *            the object type.
32   */
33  public interface AssertThatCastableObject<T> extends AssertThatObject<T> {
34  	/**
35  	 * Cast the received value into a sub-class of the receiver, before applying
36  	 * the matcher.
37  	 * 
38  	 * @param clazz
39  	 *            the target class
40  	 * @return {@link AssertThatObject the assertion on the result of the
41  	 *         conversion}.
42  	 * @param <P>
43  	 *            The subtype for the conversion
44  	 */
45  	<P extends T> AssertThatObject<P> as(Class<P> clazz);
46  
47  	/**
48  	 * Convert a received object into another one, based on a passed converter.
49  	 * 
50  	 * @param targetClass
51  	 *            he target class
52  	 * @param converter
53  	 *            the converter to be used to converter the object to the target
54  	 *            class.
55  	 * @return {@link AssertThatCastableObject the assertion on the result of
56  	 *         the conversion}.
57  	 * @param <P>
58  	 *            the target type
59  	 * @since 0.4.0
60  	 */
61  	<P> AssertThatCastableObject<P> as(Class<P> targetClass,
62  			Function<T, P> converter);
63  }
