20  package ch.powerunit.comparator.lang;
21  
22  import java.util.Comparator;
23  
24  /**
25   * First step of the DSL - Set the instance under test if necessary.
26   * 
27   * @author borettim
28   * @since 0.3.0
29   * 
30   * @param <O>
31   *            The object of the comparator
32   * @param <C>
33   *            The comparator undertest
34   */
35  public interface ComparatorTesterDSLStart<O, C extends Comparator<O>> extends
36  		ComparatorTesterDSLLess<O, C> {
37  	/**
38  	 * This method may be use to specify the instance to be used for the test of
39  	 * the comparator.
40  	 * <p>
41  	 * When not used, the framework try to create itself an instance of the
42  	 * comparator.
43  	 * 
44  	 * @param instance
45  	 *            The instance to be used for the test.
46  	 * @return {@link ComparatorTesterDSLEnd the continuation of the DSL}
47  	 * @throws NullPointerException
48  	 *             when instance is null.
49  	 */
50  	ComparatorTesterDSLLess<O, C> usingInstance(C instance);
51  }
