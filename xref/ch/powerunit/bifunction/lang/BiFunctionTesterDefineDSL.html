20  package ch.powerunit.bifunction.lang;
21  
22  import java.util.function.Supplier;
23  
24  import org.hamcrest.Matcher;
25  
26  /**
27   * Specify the expected result.
28   * 
29   * @author borettim
30   * @since 0.3.0
31   */
32  public interface BiFunctionTesterDefineDSL<T, U, R> {
33  	/**
34  	 * Specify the expected result.
35  	 * 
36  	 * @param result
37  	 *            the expected value.
38  	 * @return the {@link BiFunctionTesterNextDSL next step of the DSL}
39  	 */
40  	BiFunctionTesterNextDSL<T, U, R> thenExpectingResult(R result);
41  
42  	/**
43  	 * Specify the expected result.
44  	 * 
45  	 * @param result
46  	 *            a supplier for the expected value
47  	 * @return the {@link BiFunctionTesterNextDSL next step of the DSL}
48  	 */
49  	BiFunctionTesterNextDSL<T, U, R> thenExpectingResult(Supplier<R> result);
50  
51  	/**
52  	 * Specify the expected result.
53  	 * 
54  	 * @param matching
55  	 *            a matcher on the expected value.
56  	 * @return the {@link BiFunctionTesterNextDSL next step of the DSL}
57  	 */
58  	BiFunctionTesterNextDSL<T, U, R> thenExpectingResultThat(
59  			Matcher<? super R> matching);
60  
61  	/**
62  	 * Specify the expected result.
63  	 * 
64  	 * @param matching
65  	 *            a supplier for the matcher on the expected value
66  	 * @return the {@link BiFunctionTesterNextDSL next step of the DSL}
67  	 */
68  	BiFunctionTesterNextDSL<T, U, R> thenExpectingResultThat(
69  			Supplier<Matcher<? super R>> matching);
70  }
