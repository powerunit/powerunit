20  package ch.powerunit.impl.validator;
21  
22  import java.util.HashSet;
23  import java.util.Set;
24  
25  import javax.annotation.processing.ProcessingEnvironment;
26  import javax.annotation.processing.RoundEnvironment;
27  import javax.lang.model.element.Element;
28  import javax.lang.model.element.ElementKind;
29  import javax.lang.model.element.Modifier;
30  import javax.lang.model.element.Name;
31  import javax.lang.model.element.TypeElement;
32  import javax.lang.model.type.DeclaredType;
33  import javax.lang.model.type.TypeKind;
34  import javax.lang.model.type.TypeMirror;
35  
36  import ch.powerunit.Parameter;
37  
38  public interface ParameterProcessorValidator extends ProcessValidator {
39      default void parameterAnnotationValidation(
40              ProcessingEnvironment processingEnv, RoundEnvironment roundEnv) {
41          TypeElement predicate = processingEnv.getElementUtils().getTypeElement(
42                  "java.util.function.BiFunction");
43          Set<Name> exists = new HashSet<>();
44          roundEnv.getElementsAnnotatedWith(Parameter.class)
45                  .forEach(
46                          (Element element) -> {
47                              if (element.getKind() != ElementKind.FIELD) {
48                                  error("@Parameter must prefix a field -- "
49                                          + element + " is not a field");
50                                  return;
51                              }
52  
53                              Element parent = element.getEnclosingElement();
54                              TypeElement pp = (TypeElement) parent;
55  
56                              if (element.getAnnotation(Parameter.class).filter()) {
57                                  if (exists.contains(pp.getQualifiedName())) {
58                                      warn("Class "
59                                              + elementAsString(parent)
60                                              + "\n\t contains more than one @Parameter field with filter = true\n\tOnly one @Parameter field can use the filter attribute with true value.");
61                                  }
62                                  exists.add(pp.getQualifiedName());
63                                  TypeMirror rt = element.asType();
64                                  if (rt.getKind() != TypeKind.DECLARED) {
65                                      warn("Field "
66                                              + elementAsString(element)
67                                              + "\n\tis prefixed with @Parameter(filter=true) and is "
68                                              + element
69                                              + "\n\tA predicate field must be "
70                                              + predicate);
71                                  } else {
72                                      DeclaredType dt = (DeclaredType) rt;
73                                      if (!processingEnv.getTypeUtils()
74                                              .isSubtype(dt.asElement().asType(),
75                                                      predicate.asType())) {
76                                          warn("Field "
77                                                  + elementAsString(element)
78                                                  + "\n\tis prefixed with @Parameter(filter=true) and is "
79                                                  + dt.asElement().asType()
80                                                  + "\n\tA predicate field must be "
81                                                  + predicate);
82                                      }
83                                  }
84                              }
85                              if (element.getModifiers()
86                                      .contains(Modifier.STATIC)) {
87                                  warn("Field "
88                                          + elementAsString(element)
89                                          + "\n\tis prefixed with @Parameter and is static\n\tA parameter field can't be static");
90                              }
91                              if (!element.getModifiers().contains(
92                                      Modifier.PUBLIC)) {
93                                  warn("Field "
94                                          + elementAsString(element)
95                                          + "\n\tis prefixed with @Parameter and is not public\n\tA parameter field must be public");
96                              }
97                              if (element.getAnnotation(Parameter.class).value() < 0) {
98                                  warn("Field "
99                                          + elementAsString(element)
100                                         + "\n\tis prefixed with @Parameter and value is "
101                                         + element
102                                                 .getAnnotation(Parameter.class)
103                                                 .value()
104                                         + "\n\tA parameter field value must be >=0");
105                             }
106                         });
107     }
108 }
