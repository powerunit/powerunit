20  package ch.powerunit.impl.validator;
21  
22  import java.util.HashSet;
23  import java.util.Set;
24  
25  import javax.annotation.processing.ProcessingEnvironment;
26  import javax.annotation.processing.RoundEnvironment;
27  import javax.lang.model.element.Element;
28  import javax.lang.model.element.ElementKind;
29  import javax.lang.model.element.ExecutableElement;
30  import javax.lang.model.element.Modifier;
31  import javax.lang.model.element.TypeElement;
32  import javax.lang.model.type.DeclaredType;
33  import javax.lang.model.type.TypeKind;
34  import javax.lang.model.type.TypeMirror;
35  
36  import ch.powerunit.Parameters;
37  import ch.powerunit.TestDelegator;
38  
39  public interface ParametersProcessorValidator extends ProcessValidator {
40      default void parametersValidation(ProcessingEnvironment processingEnv,
41              RoundEnvironment roundEnv) {
42          Set<Element> exists = new HashSet<>();
43          Set<? extends Element> elements = roundEnv
44                  .getElementsAnnotatedWith(Parameters.class);
45          TypeElement stream = processingEnv.getElementUtils().getTypeElement(
46                  "java.util.stream.Stream");
47          for (Element element : elements) {
48              if (element.getKind() != ElementKind.METHOD) {
49                  error("@Parameters must prefix a method -- " + element
50                          + " is not a method");
51                  continue;
52              }
53              Element parent = element.getEnclosingElement();
54              if (exists.contains(parent)) {
55                  warn("Class "
56                          + elementAsString(parent)
57                          + "\n\t contains more than one @Parameters method\n\tOnly one @Parameters method is allowed for a test class");
58              }
59              exists.add(parent);
60              ExecutableElement ee = (ExecutableElement) element;
61              if (!ee.getModifiers().contains(Modifier.STATIC)) {
62                  warn("Method "
63                          + elementAsString(element)
64                          + "\n\tis prefixed with @Parameters and is not static\n\tThe parameters method must be static");
65              }
66              if (!ee.getModifiers().contains(Modifier.PUBLIC)) {
67                  warn("Method "
68                          + elementAsString(element)
69                          + "\n\tis prefixed with @Parameters and is not public\n\tThe parameters method must be public");
70              }
71              TypeMirror rt = ee.getReturnType();
72              if (rt.getKind() != TypeKind.DECLARED) {
73                  warn("Method " + elementAsString(element)
74                          + "\n\tis prefixed with @Parameters and return " + rt
75                          + "\n\tThe parameters method must return " + stream);
76              } else {
77                  DeclaredType dt = (DeclaredType) rt;
78                  if (!processingEnv.getTypeUtils().isSubtype(
79                          dt.asElement().asType(), stream.asType())) {
80                      warn("Method " + elementAsString(element)
81                              + "\n\tis prefixed with @Parameters and return "
82                              + dt.asElement().asType()
83                              + "\n\tThe parameters method must return " + stream);
84                  }
85              }
86              if (!ee.getParameters().isEmpty()
87                      && ee.getEnclosingElement().getAnnotation(
88                              TestDelegator.class) == null) {
89                  warn("Method "
90                          + elementAsString(element)
91                          + "\n\tis prefixed with @Parameters and is not 0-args\n\tThe parameters method must be 0-args");
92              }
93              if (ee.getParameters().size() != 1
94                      && ee.getEnclosingElement().getAnnotation(
95                              TestDelegator.class) != null) {
96                  warn("Method "
97                          + elementAsString(element)
98                          + "\n\tis prefixed with @Parameters and is not 1-args\n\tThe parameters method must be 1-args when used with TestDelegator.");
99              }
100         }
101     }
102 }
