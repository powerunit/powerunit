54  	public DefaultTestResultListener(String targetFolder,
55  			PrintStream outputConsole, PowerUnit mbean) {
56  		this.outputConsole = outputConsole;
57  		this.targetFolder = new File(targetFolder);
58  		this.targetFolder.mkdirs();
59  		this.mbean = mbean;
60  	}

129 	@Override
130 	public void notifySetStart(String setName, String parameters) {
131 		Testsuites tss = new Testsuites();
132 		tss.setDisabled(0);
133 		tss.setErrors(0);
134 		tss.setFailures(0);
135 		tss.setName(setName);
136 		tss.setTests(0);
137 		tss.setTime(0L);
138 		results.put(setName, tss);
139 
140 		Testsuite ts = new Testsuite();
141 		result.put(setName, ts);
142 		ts.setName(setName.replace('$', '.'));
143 		ts.setDisabled(0);
144 		ts.setErrors(0);
145 		ts.setFailures(0);
146 		ts.setTests(0);
147 		ts.setTime(0L);
148 		resultcase.put(setName, new HashMap<>());
149 
150 	}
151 
152 	@Override
153 	public void notifySetEnd(String setName, String parameters) {
154 		try {
155 			File target = new File(targetFolder, setName + ".xml");
156 			printf("Pushing test results into %1$s%n", target.getAbsolutePath());
157 			Object o = results.get(setName);
158 			if (results.get(setName).getTestsuite().isEmpty()) {
159 				o = result.get(setName);
160 			}
161 			if (mbean != null) {
162 				ByteArrayOutputStream str = new ByteArrayOutputStream();
163 				JAXB_CONTEXT.createMarshaller().marshal(o, str);
164 				mbean.addResult(new String(str.toByteArray()));
165 			}
166 			JAXB_CONTEXT.createMarshaller().marshal(o, target);
167 		} catch (JAXBException e) {
168 			throw new IllegalArgumentException("Unable to setup jaxb "
169 					+ e.getMessage(), e);
170 		}
171 	}

173 	@Override
174 	public void notifyStart(TestContext<T> context) {
175 		String setName = context.getSetName()
176 				+ (context.getParameterName() == null ? "" : context
177 						.getParameterName());
178 		Testsuite ts = result.get(setName);
179 		Testcase tc = new Testcase();
180 		tc.setName(context.getLocalTestName()
181 				+ (context.getParameterName() == null ? "" : "["
182 						+ context.getParameterName() + "]"));
183 		tc.setClassname(context.getTestSuiteObject().getClass()
184 				.getCanonicalName());
185 		resultcase.get(setName).put(context.getFullTestName(), tc);
186 		ts.getTestcase().add(tc);
187 		tc.setTime(System.currentTimeMillis());
188 		printf("Start of test %1$s%n", context.getFullTestName());
189 	}
190 
191 	@Override
192 	public void notifySuccess(TestContext<T> context) {
193 		String setName = context.getSetName()
194 				+ (context.getParameterName() == null ? "" : context
195 						.getParameterName());
196 		long end = System.currentTimeMillis();
197 		Testsuite ts = result.get(setName);
198 		Testcase tc = resultcase.get(setName).get(context.getFullTestName());
199 		tc.setTime((end - tc.getTime()) / 1000);
200 		ts.setTime(ts.getTime() + tc.getTime());
201 		ts.setTests(ts.getTests() + 1);
202 		printf("Success of test %1$s%n", context.getFullTestName());
203 		resumedSucess.append("\t").append(context.getLocalTestName())
204 				.append(" of ").append(context.getSetName()).append("\n");
205 	}
206 
207 	@Override
208 	public void notifyFailure(TestContext<T> context, Throwable cause) {
209 		String setName = context.getSetName()
210 				+ (context.getParameterName() == null ? "" : context
211 						.getParameterName());
212 		long end = System.currentTimeMillis();
213 		Testsuite ts = result.get(setName);
214 		Testcase tc = resultcase.get(setName).get(context.getFullTestName());
215 		tc.setTime((end - tc.getTime()) / 1000);
216 		ts.setTime(ts.getTime() + tc.getTime());
217 		error = true;
218 		ts.setFailures(ts.getFailures() + 1);
219 
220 		Failure f = new Failure();
221 		tc.getFailure().add(f);
222 		f.setType(cause.getClass().getCanonicalName());
223 		f.setMessage(cause.getMessage());
224 		StringBuilder stack = new StringBuilder("" + cause.getMessage())
225 				.append('\n');
226 		completeStack(cause, stack);
227 		f.setContent(stack.toString());
228 		printf("Failure of test %1$s because of %2$s%n",
229 				context.getFullTestName(), cause.getMessage());
230 		resumedFailure.append("\t").append(context.getLocalTestName())
231 				.append(" of ").append(context.getSetName())
232 				.append(" caused by ").append(cause.getMessage()).append("\n");
233 	}

237 		String setName = context.getSetName()
238 				+ (context.getParameterName() == null ? "" : context
239 						.getParameterName());
240 		long end = System.currentTimeMillis();
241 		Testsuite ts = result.get(setName);
242 		Testcase tc = resultcase.get(setName).get(context.getFullTestName());
243 		tc.setTime((end - tc.getTime()) / 1000);
244 		ts.setTime(ts.getTime() + tc.getTime());
245 		ts.setDisabled(ts.getDisabled() + 1);
246 
247 		tc.setSkipped("Skipped");
248 		printf("Skip of test %1$s%n", context.getFullTestName());
249 		resumedSkipped.append("\t").append(context.getLocalTestName())
250 				.append(" of ").append(context.getSetName()).append("\n");
251 	}

253 	@Override
254 	public void notifyError(TestContext<T> context, Throwable cause) {
255 		String setName = context.getSetName()
256 				+ (context.getParameterName() == null ? "" : context
257 						.getParameterName());
258 		long end = System.currentTimeMillis();
259 		Testsuite ts = result.get(setName);
260 		Testcase tc = resultcase.get(setName).get(context.getFullTestName());
261 		tc.setTime((end - tc.getTime()) / 1000);
262 		ts.setTime(ts.getTime() + tc.getTime());
263 		error = true;
264 		ts.setErrors(ts.getErrors() + 1);
265 
266 		Error e = new Error();
267 		tc.getError().add(e);
268 		e.setType(cause.getClass().getCanonicalName());
269 		e.setMessage(cause.getMessage());
270 		StringBuilder stack = new StringBuilder("" + cause.getMessage())
271 				.append('\n');
272 		completeStack(cause, stack);
273 		e.setContent(stack.toString());
274 		printf("Error of test %1$s because of %2$s%n",
275 				context.getFullTestName(), cause.getMessage());
276 		resumedFailure.append("\t").append(context.getLocalTestName())
277 				.append(" of ").append(context.getSetName())
278 				.append(" caused by ").append(cause.getMessage()).append("\n");
279 	}

281 	/**
282 	 * @param cause
283 	 * @param stack
284 	 */
285 	private void completeStack(Throwable cause, StringBuilder stack) {
286 		for (StackTraceElement ste : cause.getStackTrace()) {
287 			stack.append(ste.toString()).append('\n');
288 		}
289 		Throwable by = cause.getCause();
290 		if (by != null) {
291 			stack.append("caused by ").append(by.getMessage() + "\n");
292 			completeStack(by, stack);
293 		}
294 	}

296 	@Override
297 	public void notifyParameterStart(String setName, String parameterName) {
298 		Testsuites tss = results.get(setName);
299 		Testsuite ts = new Testsuite();
300 		result.put(setName + parameterName, ts);
301 		ts.setName(setName + "[" + parameterName + "]");
302 		ts.setDisabled(0);
303 		ts.setErrors(0);
304 		ts.setFailures(0);
305 		ts.setTests(0);
306 		ts.setTime(0L);
307 		resultcase.put(setName + parameterName, new HashMap<>());
308 		tss.getTestsuite().add(ts);
309 	}
310 
311 	@Override
312 	public void notifyParameterEnd(String setName, String parameterName) {
313 		Testsuites tss = results.get(setName);
314 		Testsuite ts = result.get(setName + parameterName);
315 		tss.setDisabled(tss.getDisabled() + ts.getDisabled());
316 		tss.setErrors(tss.getErrors() + ts.getErrors());
317 		tss.setFailures(tss.getFailures() + ts.getFailures());
318 		tss.setTests(tss.getTests() + ts.getTests());
319 		tss.setTime(tss.getTime() + ts.getTime());
320 	}
