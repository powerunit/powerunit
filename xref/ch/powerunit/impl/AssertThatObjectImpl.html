20  package ch.powerunit.impl;
21  
22  import java.util.function.Function;
23  import java.util.function.Supplier;
24  
25  import org.hamcrest.Description;
26  import org.hamcrest.Matcher;
27  import org.hamcrest.StringDescription;
28  
29  import ch.powerunit.AssertThatCastableObject;
30  import ch.powerunit.AssertThatObject;
31  import ch.powerunit.exception.AssumptionError;
32  import ch.powerunit.exception.InternalError;
33  
34  public class AssertThatObjectImpl<T> implements AssertThatObject<T>,
35  		AssertThatCastableObject<T> {
36  
37  	private final boolean assertion;
38  
39  	public AssertThatObjectImpl(boolean assertion, String msg,
40  			Supplier<T> provider) {
41  		this.provider = provider;
42  		this.msg = msg;
43  		this.assertion = assertion;
44  	}
45  
46  	private final Supplier<T> provider;
47  
48  	private final String msg;
49  
50  	@Override
51  	public boolean is(Matcher<? super T> matching) {
52  		Description message = new StringDescription();
53  		message.appendText("expecting ");
54  		matching.describeTo(message);
55  		message.appendText(" but ");
56  		T obj = provider.get();
57  		if (!matching.matches(obj)) {
58  			matching.describeMismatch(obj, message);
59  			if (assertion) {
60  				TestContextImpl<Object> ctx = DefaultPowerUnitRunnerImpl
61  						.getCurrentContext();
62  				AssertionError e = new AssertionError((msg == null ? "" : msg
63  						+ "\n")
64  						+ message.toString());
65  				if (ctx == null || ctx.isFastFail()) {
66  					throw e;
67  				} else {
68  					ctx.addAssertionError(e);
69  					return false;
70  				}
71  			} else {
72  				throw new AssumptionError((msg == null ? "" : msg + "\n")
73  						+ message.toString());
74  			}
75  		}
76  		return true;
77  	}
78  
79  	@Override
80  	public <P extends T> AssertThatObject<P> as(Class<P> clazz) {
81  		if (clazz == null) {
82  			throw new InternalError("clazz argument can't be null");
83  		}
84  		return new AssertThatObjectImpl<P>(assertion, msg, () -> {
85  			T v = provider.get();
86  			if (v == null) {
87  				return null;
88  			} else if (!clazz.isAssignableFrom(v.getClass())) {
89  				if (assertion) {
90  					throw new AssertionError((msg == null ? "" : msg + "\n")
91  							+ "The value " + v + " can't be casted to "
92  							+ clazz.getName());
93  				} else {
94  					throw new AssumptionError((msg == null ? "" : msg + "\n")
95  							+ "The value " + v + " can't be casted to "
96  							+ clazz.getName());
97  				}
98  			} else {
99  				return (P) v;
100 			}
101 		});
102 	}
103 
104 	@Override
105 	public <P> AssertThatCastableObject<P> as(Class<P> targetClass,
106 			Function<T, P> converter) {
107 		if (targetClass == null) {
108 			throw new InternalError("targetClass argument can't be null");
109 		}
110 		return new AssertThatObjectImpl<P>(assertion, msg,
111 				() -> converter.apply(provider.get()));
112 	}
113 }
