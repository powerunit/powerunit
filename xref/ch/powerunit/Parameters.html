20  package ch.powerunit;
21  
22  import java.lang.annotation.Documented;
23  import java.lang.annotation.ElementType;
24  import java.lang.annotation.Retention;
25  import java.lang.annotation.RetentionPolicy;
26  import java.lang.annotation.Target;
27  
28  /**
29   * Used to mark a method (public static, returning Stream&lt;Object[]&gt;,
30   * 0-args) as method to provide test parameter.
31   * <p>
32   * This method will be run once only.
33   * <p>
34   * For instance, we may wrote this code
35   *
36   * <pre>
37   * import java.util.Arrays;
38   * import java.util.function.Function;
39   * import java.util.stream.Stream;
40   * 
41   * import ch.powerunit.Parameter;
42   * import ch.powerunit.Parameters;
43   * import ch.powerunit.Test;
44   * import ch.powerunit.TestSuite;
45   * 
46   * public class FunctionParameterTest&lt;T, R&gt; implements TestSuite {
47   * 
48   * 	&#064;Parameters(&quot;%1$s on %2$s expecting %3$s&quot;)
49   * 	public static Stream&lt;Object[]&gt; getDatas() {
50   * 		return Arrays.stream(new Object[][] { {
51   * 				(Function&lt;String, Integer&gt;) Integer::valueOf, &quot;1&quot;, 1 } });
52   * 	}
53   * 
54   * 	&#064;Parameter(0)
55   * 	public Function&lt;T, R&gt; function;
56   * 
57   * 	&#064;Parameter(1)
58   * 	public T input;
59   * 
60   * 	&#064;Parameter(2)
61   * 	public R expected;
62   * 
63   * 	&#064;Test
64   * 	public void testAFunction() {
65   * 		assertThatFunction(function, input).is(expected);
66   * 	}
67   * }
68   *
69   * </pre>
70   * 
71   * It is also possible to indicate that each test parameter set is not
72   * applicable for each test method. This is done by using an additional, with
73   * the attribute <code>filter=true</code>. This field will be a BiFunction
74   * receiving the test method name and the parameters and returning a boolean. (
75   * <code>BiFunction&lt;String,Object[],Boolean&gt;</code>). This method will be
76   * used to check if the test method accept (or not) the parameter.
77   * 
78   * <p>
79   * In case this annotation is used on a method from a test class annotated with
80   * <code>{@link TestDelegator}</code>, this method is used as a way to provide
81   * parameter to the test classes. In this case, this method must have exactly
82   * one parameter, which will be a class annotated with
83   * <code>{@link TestInterface}</code>.
84   *
85   * @author borettim
86   * @see java.util.stream.Stream
87   * @see TestSuite#addFieldToEachEntry(Object) This is used on the stream to add
88   *      an object at the end of each entry (for instance the BiFunction).
89   */
90  @Documented
91  @Retention(RetentionPolicy.RUNTIME)
92  @Target(ElementType.METHOD)
93  public @interface Parameters {
94  	/**
95  	 * Define an optional name of the test parameters. use the syntax defined by
96  	 * {@link java.lang.String#format(String, Object...)} to refer to the
97  	 * parameter.
98  	 * <p>
99  	 * <b>The syntax for the formatted string has change since the version
100 	 * 0.4.0.</b>. At the moment, both syntax are supported, but the old one may
101 	 * be removed in a future release.
102 	 * <br>
103 	 * <h3>Normal syntax</h3>
104 	 * The syntax defined by {@link java.lang.String#format(String, Object...)}
105 	 * is used. Each parameter is available. It is here required to be careful
106 	 * with the numbering of the argument ; {@link Parameter} index are numbered
107 	 * starting from 0, but in the string format the numbering start from 1. In
108 	 * the easy form, the parameter can be passed with the syntax
109 	 * <code>%1$s</code> which for instance will use the first parameter (0
110 	 * parameter), as a String. 
111 	 * <br>
112 	 * <h3>Old syntax</h3>
113 	 * The old syntax is defined by
114 	 * {@link java.text.MessageFormat#format(String, Object...)} ; This syntax
115 	 * is enabled when the passed string format contains the regular expression
116 	 * <code>\{[0-9]+\}</code>.
117 	 * 
118 	 * @return the name.
119 	 * @see java.text.MessageFormat#format(String, Object...) The formatter used
120 	 *      for parameterized test. For version before 0.4.0.
121 	 * @see java.lang.String#format(String, Object...) The formatter used for
122 	 *      parameterized test, since version 0.4.0.
123 	 */
124 	String value() default "";
125 }
