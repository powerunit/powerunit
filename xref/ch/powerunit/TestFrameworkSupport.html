20  package ch.powerunit;
21  
22  import java.util.Comparator;
23  import java.util.function.BiFunction;
24  import java.util.function.Function;
25  import java.util.regex.Pattern;
26  
27  import org.hamcrest.Matcher;
28  
29  import ch.powerunit.bifunction.BiFunctionTester;
30  import ch.powerunit.bifunction.lang.BiFunctionTesterStartDSL;
31  import ch.powerunit.comparator.ComparatorTester;
32  import ch.powerunit.comparator.lang.ComparatorTesterDSLStart;
33  import ch.powerunit.function.FunctionTester;
34  import ch.powerunit.function.lang.FunctionTesterStartDSL;
35  import ch.powerunit.matchers.MatcherTester;
36  import ch.powerunit.matchers.lang.MatcherTesterDSL1;
37  import ch.powerunit.pattern.PatternTester;
38  import ch.powerunit.pattern.lang.PatternTester0;
39  
40  /**
41   * This is the interface for the test support.
42   * 
43   * @author borettim
44   * @since 0.3.0
45   */
46  interface TestFrameworkSupport {
47  
48  	/**
49  	 * Start the creation of a tester definition for a matcher.
50  	 * <p>
51  	 * <b>{@link ch.powerunit.matchers.MatcherTester#of(Class) Please refer to
52  	 * the complete documentation}</b>
53  	 * 
54  	 * @param matcherClass
55  	 *            the class of the Matcher.
56  	 * @return {@link ch.powerunit.matchers.lang.MatcherTesterDSL1 The DSL}
57  	 * @see ch.powerunit.matchers.MatcherTester#of(Class)
58  	 * @see ch.powerunit.TestDelegate
59  	 * @since 0.3.0
60  	 * @param <T>
61  	 *            The matcher class
62  	 */
63  	default <T extends Matcher<?>> MatcherTesterDSL1<T> testerOfMatcher(
64  			Class<T> matcherClass) {
65  		return MatcherTester.of(matcherClass);
66  	}
67  
68  	/**
69  	 * Use this method to start the DSL to test a comparator.
70  	 * <p>
71  	 * <b>{@link ch.powerunit.comparator.ComparatorTester#of(Class) Please refer
72  	 * to the complete documentation}</b>
73  	 * 
74  	 * @param comparatorClass
75  	 *            the class of the Comparator
76  	 * @return {@link ch.powerunit.comparator.lang.ComparatorTesterDSLStart the
77  	 *         DSL}
78  	 * @see ch.powerunit.comparator.ComparatorTester#of(Class)
79  	 * @see ch.powerunit.TestDelegate
80  	 * @since 0.3.0
81  	 * @param <O>
82  	 *            The object of the comparator
83  	 * @param <C>
84  	 *            The comparator undertest
85  	 */
86  	default <O, C extends Comparator<O>> ComparatorTesterDSLStart<O, C> testerOfComparator(
87  			Class<C> comparatorClass) {
88  		return ComparatorTester.of(comparatorClass);
89  	}
90  
91  	/**
92  	 * Use this method to start the DSL to test a function.
93  	 * <p>
94  	 * <b>{@link ch.powerunit.function.FunctionTester#of(Function) Please refer
95  	 * to the complete documentation}</b>
96  	 * 
97  	 * @param functionUnderTest
98  	 *            the function to be tested
99  	 * @return {@link ch.powerunit.function.lang.FunctionTesterStartDSL the DSL}
100 	 * @see ch.powerunit.function.FunctionTester#of(Function)
101 	 * @see ch.powerunit.TestDelegate
102 	 * @since 0.3.0
103 	 * @param <T>
104 	 *            The input argument type
105 	 * @param <R>
106 	 *            The result type
107 	 */
108 	default <T, R> FunctionTesterStartDSL<T, R> testerOfFunction(
109 			Function<T, R> functionUnderTest) {
110 		return FunctionTester.of(functionUnderTest);
111 	}
112 
113 	/**
114 	 * Use this method to start the DSL to test a bifunction.
115 	 * <p>
116 	 * <b>{@link ch.powerunit.bifunction.BiFunctionTester#of(BiFunction) Please
117 	 * refer to the complete documentation}</b>
118 	 * 
119 	 * @param bifunctionUnderTest
120 	 *            the bifunction to be tested
121 	 * @return {@link ch.powerunit.bifunction.lang.BiFunctionTesterStartDSL the
122 	 *         DSL}
123 	 * @see ch.powerunit.bifunction.BiFunctionTester#of(BiFunction)
124 	 * @see ch.powerunit.TestDelegate
125 	 * @since 0.3.0
126 	 * @param <T>
127 	 *            The first input argument type
128 	 * @param <U>
129 	 *            the second input argument type
130 	 * @param <R>
131 	 *            The result type
132 	 */
133 	default <T, U, R> BiFunctionTesterStartDSL<T, U, R> testerOfBiFunction(
134 			BiFunction<T, U, R> bifunctionUnderTest) {
135 		return BiFunctionTester.of(bifunctionUnderTest);
136 	}
137 
138 	/**
139 	 * Start the DSL to create a tester of Pattern, based on a String.
140 	 * <p>
141 	 * The passed String will be compiled as a Pattern. <br>
142 	 * For instance :
143 	 * 
144 	 * <pre>
145 	 * &#064;TestDelegate
146 	 * public final PatternTester sample1 = testerOfPattern(&quot;a+&quot;).receiving(&quot;b&quot;)
147 	 * 		.thenNoMatching().receiving(&quot;aa&quot;).thenMatching().build();
148 	 * </pre>
149 	 * 
150 	 * @param pattern
151 	 *            the pattern, as a String
152 	 * @return {@link PatternTester0#receiving(String) The next step of the
153 	 *         DSL.}
154 	 * @see #testerOfPattern(Pattern)
155 	 * @see ch.powerunit.pattern.PatternTester
156 	 * @since 0.4.0
157 	 */
158 	default PatternTester0 testerOfPattern(String pattern) {
159 		return PatternTester.of(pattern);
160 	}
161 
162 	/**
163 	 * Start the DSL to create a tester of Pattern, based on a String.
164 	 * <p>
165 	 * For instance :
166 	 * 
167 	 * <pre>
168 	 * &#064;TestDelegate
169 	 * public final PatternTester sample1 = testerOfPattern(Pattern.compile(&quot;a+&quot;))
170 	 * 		.receiving(&quot;b&quot;).thenNoMatching().receiving(&quot;aa&quot;).thenMatching().build();
171 	 * </pre>
172 	 * 
173 	 * @param pattern
174 	 *            the pattern.
175 	 * @return {@link PatternTester0#receiving(String) The next step of the
176 	 *         DSL.}
177 	 * @see ch.powerunit.pattern.PatternTester
178 	 * @since 0.4.0
179 	 */
180 	default PatternTester0 testerOfPattern(Pattern pattern) {
181 		return PatternTester.of(pattern);
182 	}
183 }
