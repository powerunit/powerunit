50  @SupportedAnnotationTypes({ "org.powerunit.Test", "org.powerunit.Rule",
51  		"org.powerunit.Parameters", "org.powerunit.Parameter",
52  		"org.powerunit.Ignore", "org.powerunit.Categories" })
53  @SupportedSourceVersion(SourceVersion.RELEASE_8)
54  public class PowerUnitProcessor extends AbstractProcessor implements
55  		ParametersProcessorValidator, ParameterProcessorValidator,
56  		RuleProcessorValidator, TestProcessorValidator,
57  		IgnoreProcessorValidator {
58  
59  	private boolean categoryDone = false;
60  
61  	@Override
62  	public boolean process(Set<? extends TypeElement> annotations,
63  			RoundEnvironment roundEnv) {
64  		if (!roundEnv.processingOver()) {
65  			testAnnotationValidation(roundEnv);
66  			ruleAnnotationValidation(processingEnv, roundEnv);
67  			parametersValidation(processingEnv, roundEnv);
68  			parameterAnnotationValidation(roundEnv);
69  			ignoreAnnotationValidation(roundEnv);
70  			try {
71  				if (!categoryDone) {
72  					generateCategorySuite(roundEnv);
73  				}
74  			} catch (IOException e) {
75  				error("Unable to generate test suite file, because of "
76  						+ e.getMessage());
77  			}
78  		}
79  
80  		return true;
81  	}
82  
83  	private void generateCategorySuite(RoundEnvironment roundEnv)
84  			throws IOException {
85  		categoryDone = true;
86  		Set<? extends Element> elements = roundEnv
87  				.getElementsAnnotatedWith(Categories.class);
88  		Map<String, List<TypeMirror>> categories = new HashMap<>();
89  		for (Element element : elements) {
90  			if (element.getKind() == ElementKind.CLASS) {
91  				TypeMirror current = element.asType();
92  				Categories ca = element.getAnnotation(Categories.class);
93  				if (ca != null) {
94  					String cat[] = ca.value();
95  					for (String c : cat) {
96  						List<TypeMirror> lst = categories.get(c);
97  						if (lst == null) {
98  							lst = new ArrayList<>();
99  						}
100 						lst.add(current);
101 						categories.put(c, lst);
102 					}
103 				}
104 			}
105 		}
106 		JavaFileObject suite = processingEnv.getFiler().createSourceFile(
107 				"org.powerunit.suite.Suites",
108 				elements.toArray(new Element[] {}));
109 		try (PrintWriter w = new PrintWriter(suite.openWriter());) {
110 			w.println("package org.powerunit.suite;");
111 			w.println();
112 			w.println("@javax.annotation.Generated(\"Test suites for powerunit\")");
113 			w.println("public final class Suites implements org.powerunit.TestResultListener {");
114 			w.println();
115 			w.println("  private Suites(String targetFolder) {");
116 			w.println("    this.targetFolder=new java.io.File(targetFolder);");
117 			w.println("    this.targetFolder.mkdirs();");
118 			w.println("  }");
119 			w.println();
120 			w.println("  private final static javax.xml.bind.JAXBContext context;");
121 			w.println();
122 			w.println("  static {");
123 			w.println("    try {");
124 			w.println("      context = javax.xml.bind.JAXBContext.newInstance(org.powerunit.report.Testsuites.class);");
125 			w.println("    } catch (javax.xml.bind.JAXBException e) {");
126 			w.println("      throw new java.lang.IllegalArgumentException(\"Unable to setup jaxb \"+ e.getMessage(), e);");
127 			w.println("    }");
128 			w.println("  }");
129 			w.println();
130 			w.println("  private static final java.util.Map<String,Class<?>[]> suites;");
131 			w.println();
132 			w.println("  static {");
133 			w.println("    java.util.Map<String,Class<?>[]> tmp = new java.util.HashMap<>();");
134 			for (Map.Entry<String, List<TypeMirror>> cat : categories
135 					.entrySet()) {
136 				w.println("    tmp.put(\"" + cat.getKey()
137 						+ "\",new Class<?>[]{");
138 				int counter = 0;
139 				for (TypeMirror tm : cat.getValue()) {
140 					if (counter > 0) {
141 						w.println(",");
142 					}
143 					w.print("      "
144 							+ ((DeclaredType) tm).asElement().toString()
145 							+ ".class");
146 					counter++;
147 				}
148 				w.println();
149 				w.println("    });");
150 			}
151 			w.println("    suites=java.util.Collections.unmodifiableMap(tmp);");
152 			w.println("  }");
153 			w.println();
154 			w.println("  public static java.util.Map<String,Class<?>[]> getSuites() {");
155 			w.println("    return suites;");
156 			w.println("  }");
157 			w.println();
158 			w.println("  private boolean error=false;");
159 			w.println();
160 			w.println("  private java.util.Map<String,org.powerunit.report.Testsuite> result = new java.util.HashMap<>();");
161 			w.println();
162 			w.println("  private java.io.File targetFolder ;");
163 			w.println();
164 			w.println("  @Override");
165 			w.println("  public void notifySetStart(String setName, String parameters) {");
166 			w.println("    System.out.println(\"Start of \"+setName);");
167 			w.println("    result.put(setName, new org.powerunit.report.Testsuite());");
168 			w.println("    result.get(setName).setName(setName.replace('$','.'));");
169 			w.println("    result.get(setName).setDisabled(0);");
170 			w.println("    result.get(setName).setErrors(0);");
171 			w.println("    result.get(setName).setFailures(0);");
172 			w.println("    result.get(setName).setTests(0);");
173 			w.println("  }");
174 			w.println();
175 			w.println("  @Override");
176 			w.println("  public void notifySetEnd(String setName, String parameters) {");
177 			w.println("    System.out.println(\"End of \"+setName);");
178 			w.println("    try {");
179 			w.println("      java.io.File target = new java.io.File(targetFolder,setName+\".xml\");");
180 			w.println("      context.createMarshaller().marshal(result.get(setName), target);");
181 			w.println("    } catch (javax.xml.bind.JAXBException e) {");
182 			w.println("      e.printStackTrace();");
183 			w.println("      System.exit(1);");
184 			w.println("    }");
185 			w.println("  }");
186 			w.println();
187 			w.println("  @Override");
188 			w.println("  public void notifyStart(org.powerunit.TestContext context) {");
189 			w.println("    System.out.println(\"Start of \"+context.getFullTestName());");
190 			w.println("  }");
191 			w.println();
192 			w.println("  @Override");
193 			w.println("  public void notifySuccess(org.powerunit.TestContext context) {");
194 			w.println("    System.out.println(\"End of \"+context.getFullTestName());");
195 			w.println("    result.get(context.getSetName()).setTests(result.get(context.getSetName()).getTests()+1);");
196 			w.println("    result.get(context.getSetName()).getTestcase().add(new org.powerunit.report.Testcase());");
197 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setName(context.getLocalTestName()+(context.getParameterName()==null?\"\":(\"[\"+context.getParameterName()+\"]\")));");
198 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setClassname(context.getTestSuiteObject().getClass().getCanonicalName());");
199 			w.println("  }");
200 			w.println();
201 			w.println("  @Override");
202 			w.println("  public void notifyFailure(org.powerunit.TestContext context, Throwable cause) {");
203 			w.println("    System.out.println(\"Failure of \"+context.getFullTestName()+\" because of \"+cause.getMessage());");
204 			w.println("    cause.printStackTrace();");
205 			w.println("    error=true;");
206 			w.println("    result.get(context.getSetName()).setFailures(result.get(context.getSetName()).getFailures()+1);");
207 			w.println("    result.get(context.getSetName()).getTestcase().add(new org.powerunit.report.Testcase());");
208 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setName(context.getLocalTestName()+(context.getParameterName()==null?\"\":(\"[\"+context.getParameterName()+\"]\")));");
209 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setClassname(context.getTestSuiteObject().getClass().getCanonicalName());");
210 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getFailure().add(new org.powerunit.report.Failure());");
211 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getFailure().get(result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getFailure().size()-1).setType(cause.getClass().getCanonicalName());");
212 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getFailure().get(result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getFailure().size()-1).setMessage(cause.getMessage());");
213 			w.println("  }");
214 			w.println();
215 			w.println("  @Override");
216 			w.println("  public void notifySkipped(org.powerunit.TestContext context) {");
217 			w.println("    System.out.println(\"Skip of \"+context.getFullTestName());");
218 			w.println("    result.get(context.getSetName()).setDisabled(result.get(context.getSetName()).getDisabled()+1);");
219 			w.println("    result.get(context.getSetName()).getTestcase().add(new org.powerunit.report.Testcase());");
220 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setName(context.getLocalTestName()+(context.getParameterName()==null?\"\":(\"[\"+context.getParameterName()+\"]\")));");
221 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setClassname(context.getTestSuiteObject().getClass().getCanonicalName());");
222 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setSkipped(\"\");");
223 			w.println("  }");
224 			w.println();
225 			w.println("  @Override");
226 			w.println("  public void notifyError(org.powerunit.TestContext context, Throwable cause) {");
227 			w.println("    System.out.println(\"Error of \"+context.getFullTestName()+\" because of \"+cause.getMessage());");
228 			w.println("    cause.printStackTrace();");
229 			w.println("    error=true;");
230 			w.println("    result.get(context.getSetName()).setErrors(result.get(context.getSetName()).getErrors()+1);");
231 			w.println("    result.get(context.getSetName()).getTestcase().add(new org.powerunit.report.Testcase());");
232 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setName(context.getLocalTestName()+(context.getParameterName()==null?\"\":(\"[\"+context.getParameterName()+\"]\")));");
233 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).setClassname(context.getTestSuiteObject().getClass().getCanonicalName());");
234 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getError().add(new org.powerunit.report.Error());");
235 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getError().get(result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getError().size()-1).setType(cause.getClass().getCanonicalName());");
236 			w.println("    result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getError().get(result.get(context.getSetName()).getTestcase().get(result.get(context.getSetName()).getTestcase().size()-1).getError().size()-1).setMessage(cause.getMessage());");
237 			w.println("  }");
238 			w.println();
239 			w.println("  public static void main(String argv[]) {");
240 			w.println("    if (argv.length==1) {");
241 			w.println("       suites.keySet().stream().forEach((c)->{main(new String[]{argv[0],c});});");
242 			w.println("    return;");
243 			w.println("    }");
244 			w.println("    String name = argv[1];");
245 			w.println("    boolean success=true;");
246 			w.println("    for(Class<?> c:getSuites().get(name)) {");
247 			w.println("      Suites p = new Suites(argv[0]);");
248 			w.println("      org.powerunit.PowerUnitRunner r = new org.powerunit.impl.DefaultPowerUnitRunnerImpl(c);");
249 			w.println("      r.addListener(p);");
250 			w.println("      r.run();");
251 			w.println("      success=success&&!p.error;");
252 			w.println("    }");
253 			w.println("    if(!success) {");
254 			w.println("      System.out.println(\"Tests in error.\");");
255 			w.println("      System.exit(-1);");
256 			w.println("    }");
257 			w.println("  }");
258 			w.println();
259 			w.println("}");
260 		}
261 	}

263 	@Override
264 	public String elementAsString(Element ee) {
265 		return ee.toString() + " of " + ee.getEnclosingElement().toString();
266 	}

268 	@Override
269 	public void error(String msg) {
270 		processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, msg);
271 	}
272 
273 	@Override
274 	public void warn(String msg) {
275 		processingEnv.getMessager().printMessage(
276 				Diagnostic.Kind.MANDATORY_WARNING, msg);
277 	}
