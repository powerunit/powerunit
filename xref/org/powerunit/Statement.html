20  package org.powerunit;
21  
22  import java.lang.reflect.InvocationTargetException;
23  import java.lang.reflect.Method;
24  import java.util.Objects;
25  
26  import org.powerunit.exception.InternalError;
27  
28  /**
29   * Definition of a statement (piece of code that can thrown Throwable).
30   * <p>
31   * A statement can be used inside test (to isolate a code that must thrown an
32   * exception) and are used internally by the framework to compose and execute
33   * test sequence element.
34   * 
35   * @author borettim
36   * @param <P>
37   *            The type of the parameter
38   * @param <T>
39   *            the exception type
40   */
41  @FunctionalInterface
42  public interface Statement<P, T extends Throwable> {
43  
44  	/**
45  	 * Executable code.
46  	 * 
47  	 * @param parameter
48  	 *            A parameter for the statement
49  	 * @throws Throwable
50  	 *             in case of error.
51  	 */
52  	void run(P parameter) throws Throwable;// should be T, but T seem to produce
53  											// a bug in the compiler
54  
55  	/**
56  	 * Aggregate this statement and then the following. The second statement is
57  	 * done, even in case of exception in the first one.
58  	 * 
59  	 * @param after
60  	 *            the next statement
61  	 * @return the new statement
62  	 */
63  	default Statement<P, T> andThenAlways(Statement<P, T> after) {
64  		Objects.requireNonNull(after);
65  		return (p) -> {
66  			try {
67  				run(p);
68  			} finally {
69  				after.run(p);
70  			}
71  		};
72  	}
73  
74  	/**
75  	 * Aggregate this statement and then the following. The second statement is
76  	 * done except in case of exception in the first one.
77  	 * 
78  	 * @param after
79  	 *            the next statement
80  	 * @return the new statement
81  	 */
82  	default Statement<P, T> andThenOnlySuccess(Statement<P, T> after) {
83  		Objects.requireNonNull(after);
84  		return (p) -> {
85  			run(p);
86  			after.run(p);
87  		};
88  	}
89  
90  	/**
91  	 * Build a around statement (do something, then something others, and after
92  	 * one a third statement, event in case of exception.
93  	 * 
94  	 * @param internal
95  	 *            the internal part
96  	 * @param before
97  	 *            the first statement
98  	 * @param after
99  	 *            the last statement, done event in case of exception.
100 	 * @return the new statement.
101 	 * @param <P>
102 	 *            The type of the parameter
103 	 * @param <T>
104 	 *            the exception type
105 	 */
106 	static <P, T extends Throwable> Statement<P, T> around(
107 			Statement<P, T> internal, Statement<P, T> before,
108 			Statement<P, T> after) {
109 		Objects.requireNonNull(internal);
110 		Objects.requireNonNull(before);
111 		Objects.requireNonNull(after);
112 		return before.andThenOnlySuccess(internal).andThenAlways(after);
113 	}
114 
115 	/**
116 	 * Build a statement based on a method-
117 	 * 
118 	 * @param target
119 	 *            the target object
120 	 * @param method
121 	 *            the method
122 	 * @return the new statement.
123 	 * @param <P>
124 	 *            The type of the parameter
125 	 * @param <T>
126 	 *            the exception type
127 	 */
128 	static <P, T extends Throwable> Statement<P, T> reflectionMethod(
129 			Object target, Method method) {
130 		Objects.requireNonNull(target);
131 		Objects.requireNonNull(method);
132 		return (p) -> {
133 			try {
134 				method.invoke(target);
135 			} catch (InvocationTargetException e) {
136 				throw e.getCause();
137 			} catch (IllegalAccessException | IllegalArgumentException e) {
138 				throw new InternalError("Unexpected error " + e.getMessage(), e);
139 			}
140 		};
141 	}
142 
143 }
