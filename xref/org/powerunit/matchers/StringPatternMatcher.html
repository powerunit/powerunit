20  package org.powerunit.matchers;
21  
22  import java.util.regex.Pattern;
23  
24  import org.hamcrest.Description;
25  import org.hamcrest.TypeSafeDiagnosingMatcher;
26  
27  /**
28   * The matcher on string pattern.
29   * 
30   * @author borettim
31   *
32   */
33  public class StringPatternMatcher extends TypeSafeDiagnosingMatcher<String> {
34  
35  	/**
36  	 * @param pattern
37  	 */
38  	public StringPatternMatcher(Pattern pattern) {
39  		this.pattern = pattern;
40  	}
41  
42  	private Pattern pattern;
43  
44  	@Override
45  	public void describeTo(Description description) {
46  		description.appendText("a string matching the pattern ").appendValue(
47  				pattern);
48  	}
49  
50  	@Override
51  	protected boolean matchesSafely(String item, Description mismatchDescription) {
52  		if (!pattern.matcher(item).matches()) {
53  			mismatchDescription.appendText("but the string ").appendValue(item)
54  					.appendText(" was not matching ").appendValue(pattern);
55  			return false;
56  		}
57  		return true;
58  	}
59  
60  }
