20  package org.powerunit;
21  
22  import org.hamcrest.Matcher;
23  import org.hamcrest.Matchers;
24  
25  /**
26   * DSL on assertion on iterable.
27   * <p>
28   * This interface is returned by the various methods
29   * {@link TestSuite#assertThatIterable(Iterable) assertThatIterable} exposed by
30   * {@link TestSuite}.
31   * 
32   * @author borettim
33   * @param <T>
34   *            the Element type of the Iterator
35   * @see AssertThatObject assertion on object are also available on iterable.
36   */
37  public interface AssertThatIterable<T> extends AssertThatObject<Iterable<T>> {
38  	/**
39  	 * Check the size of the iterable.
40  	 * 
41  	 * @param size
42  	 *            the expected size.
43  	 */
44  	@SuppressWarnings({ "unchecked", "rawtypes" })
45  	default void hasSize(int size) {
46  		is((Matcher) Matchers.hasSize(size));
47  	}
48  
49  	/**
50  	 * Check the size is not the one passed.
51  	 * 
52  	 * @param size
53  	 *            the not expected size.
54  	 */
55  	@SuppressWarnings({ "unchecked", "rawtypes" })
56  	default void hasNotSize(int size) {
57  		isNot((Matcher) Matchers.hasSize(size));
58  	}
59  
60  	/**
61  	 * Check that the iterable contains the items.
62  	 * 
63  	 * @param items
64  	 *            the expected items.
65  	 */
66  	default void contains(@SuppressWarnings("unchecked") T... items) {
67  		is(Matchers.contains(items));
68  	}
69  
70  	/**
71  	 * Check that the iterable contains the items.
72  	 * 
73  	 * @param items
74  	 *            the matcher for each item.
75  	 */
76  	default void containsMatching(
77  			@SuppressWarnings("unchecked") Matcher<? super T>... items) {
78  		is(Matchers.contains(items));
79  	}
80  
81  	/**
82  	 * Check that the iterable contains the items in any order.
83  	 * 
84  	 * @param items
85  	 *            the expected items.
86  	 */
87  	default void containsInAnyOrder(@SuppressWarnings("unchecked") T... items) {
88  		is(Matchers.containsInAnyOrder(items));
89  	}
90  
91  	/**
92  	 * Check that the iterable contains the items in any order.
93  	 * 
94  	 * @param items
95  	 *            the matcher for each item.
96  	 */
97  	default void containsInAnyOrderMatching(
98  			@SuppressWarnings("unchecked") Matcher<? super T>... items) {
99  		is(Matchers.containsInAnyOrder(items));
100 	}
101 }
