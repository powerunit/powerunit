20  package ch.powerunit.test.core;
21  
22  import ch.powerunit.TestSuite;
23  
24  public class AssertThatFunctionTests {
25  
26      public static void main(String[] args) {
27          AllTests.testNoException(
28                  "testAssertThatFunctionOKWithoutMessage",
29                  () -> TestSuite.DSL.assertThatFunction((a) -> a + "x", "").is(
30                          "x"));
31  
32          AllTests.testException(
33                  "testAssertThatFunctionKOWithoutMessage",
34                  () -> TestSuite.DSL.assertThatFunction((a) -> a + "x", "").is(
35                          "xy"), "expecting \"xy\" but was \"x\"");
36  
37          AllTests.testNoException(
38                  "testAssertThatFunctionOKWithMessage",
39                  () -> TestSuite.DSL.assertThatFunction("msg", (a) -> a + "x",
40                          "").is("x"));
41  
42          AllTests.testException(
43                  "testAssertThatFunctionKOWithMessage",
44                  () -> TestSuite.DSL.assertThatFunction("msg", (a) -> a + "x",
45                          "").is("xy"), "msg\nexpecting \"xy\" but was \"x\"");
46      }
47  
48  }
