20  package ch.powerunit.test.core;
21  
22  import ch.powerunit.TestSuite;
23  
24  public class AssumeThatTests {
25      @SuppressWarnings({ "rawtypes", "unchecked" })
26      public static void main(String[] args) {
27          AllTests.testNoException("testAssumeThatIsObjectOKWithoutMessage",
28                  () -> TestSuite.DSL.assumeThat("x").is("x"));
29  
30          AllTests.testException("testAssumeThatIsObjectKOWithoutMessage",
31                  () -> TestSuite.DSL.assumeThat("x").is("xy"),
32                  "expecting \"xy\" but was \"x\"");
33  
34          AllTests.testNoException("testAssumeThatIsObjectOKWithMessage",
35                  () -> TestSuite.DSL.assumeThat("msg", "x").is("x"));
36  
37          AllTests.testException("testAssumeThatIsObjectKOWithMessage",
38                  () -> TestSuite.DSL.assumeThat("msg", "x").is("xy"),
39                  "msg\nexpecting \"xy\" but was \"x\"");
40  
41          AllTests.testNoException("testAssumeThatIsNotObjectOKWithoutMessage",
42                  () -> TestSuite.DSL.assumeThat("x").isNot("y"));
43  
44          AllTests.testException("testAssumeThatIsNotObjectKOWithoutMessage",
45                  () -> TestSuite.DSL.assumeThat("x").isNot("x"),
46                  "expecting not \"x\" but was \"x\"");
47  
48          AllTests.testNoException("testAssumeThatIsNotObjectOKWithMessage",
49                  () -> TestSuite.DSL.assumeThat("msg", "x").isNot("y"));
50  
51          AllTests.testException("testAssumeThatIsNotObjectKOWithMessage",
52                  () -> TestSuite.DSL.assumeThat("msg", "x").isNot("x"),
53                  "msg\nexpecting not \"x\" but was \"x\"");
54  
55          AllTests.testNoException("testAssumeThatIsNullOKWithoutMesage",
56                  () -> TestSuite.DSL.assumeThat((Object) null).isNull());
57  
58          AllTests.testException("testAssumeThatIsNullKOWithoutMesage",
59                  () -> TestSuite.DSL.assumeThat("x").isNull(),
60                  "expecting null but was \"x\"");
61  
62          AllTests.testNoException("testAssumeThatIsNullOKWithMesage",
63                  () -> TestSuite.DSL.assumeThat("msg", (Object) null).isNull());
64  
65          AllTests.testException("testAssumeThatIsNullKOWithMesage",
66                  () -> TestSuite.DSL.assumeThat("msg", "x").isNull(),
67                  "msg\nexpecting null but was \"x\"");
68  
69          AllTests.testNoException("testAssumeThatIsNotNullOKWithoutMesage",
70                  () -> TestSuite.DSL.assumeThat("x").isNotNull());
71  
72          AllTests.testException("testAssumeThatIsNotNullKOWithoutMesage",
73                  () -> TestSuite.DSL.assumeThat((Object) null).isNotNull(),
74                  "expecting not null but was null");
75  
76          AllTests.testNoException("testAssumeThatIsNotNullOKWithMesage",
77                  () -> TestSuite.DSL.assumeThat("msg", "x").isNotNull());
78  
79          AllTests.testException("testAssumeThatIsNotNullKOWithMesage",
80                  () -> TestSuite.DSL.assumeThat("msg", (Object) null)
81                          .isNotNull(), "msg\nexpecting not null but was null");
82  
83          //
84          AllTests.testNoException("testAssumeThatIsAOKWithoutMesage",
85                  () -> TestSuite.DSL.assumeThat("x").isA(String.class));
86  
87          AllTests.testException("testAssumeThatIsAKOWithoutMesage",
88                  () -> TestSuite.DSL.assumeThat("x").isA((Class) Integer.class),
89                  "expecting is an instance of java.lang.Integer but \"x\" is a java.lang.String");
90  
91          AllTests.testNoException("testAssumeThatIsAOKWithMesage",
92                  () -> TestSuite.DSL.assumeThat("msg", "x").isA(String.class));
93  
94          AllTests.testException(
95                  "testAssumeThatIIsAKOWithMesage",
96                  () -> TestSuite.DSL.assumeThat("msg", "x").isA(
97                          (Class) Integer.class),
98                  "msg\nexpecting is an instance of java.lang.Integer but \"x\" is a java.lang.String");
99      }
