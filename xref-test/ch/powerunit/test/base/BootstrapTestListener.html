20  package ch.powerunit.test.base;
21  
22  import ch.powerunit.TestContext;
23  import ch.powerunit.TestResultListener;
24  import ch.powerunit.TestSuite;
25  
26  class BootstrapTestListener<T> implements TestResultListener<T>, TestSuite {// package
27                                                                              // private
28  
29      @Override
30      public void notifySetStart(String setName, String parameters) {
31          System.out.println("Starting " + setName + ":" + parameters);
32      }
33  
34      @Override
35      public void notifySetEnd(String setName, String parameters) {
36          System.out.println("Ending " + setName + ":" + parameters);
37      }
38  
39      @Override
40      public void notifyStart(TestContext<T> context) {
41          System.out.println("Starting " + context.getFullTestName() + ":"
42                  + context.getTestCategories());
43      }
44  
45      @Override
46      public void notifySuccess(TestContext<T> context) {
47          System.out.println("Successs " + context.getFullTestName() + ":"
48                  + context.getTestCategories());
49      }
50  
51      @Override
52      public void notifyFailure(TestContext<T> context, Throwable cause) {
53          System.out.println("Failed " + context.getFullTestName() + ":"
54                  + context.getTestCategories());
55          fail("Unexpected error " + cause.getMessage(), cause);
56      }
57  
58      @Override
59      public void notifySkipped(TestContext<T> context) {
60          System.out.println("Skipped " + context.getFullTestName() + ":"
61                  + context.getTestCategories());
62      }
63  
64      @Override
65      public void notifyError(TestContext<T> context, Throwable cause) {
66          System.out.println("Failed " + context.getFullTestName() + ":"
67                  + context.getTestCategories());
68          fail("Unexpected error " + cause.getMessage(), cause);
69      }
70  
71      @Override
72      public void notifyParameterStart(String setName, String parameterName) {
73          System.out.println("Start of parameter " + parameterName + " on "
74                  + setName);
75      }
76  
77      @Override
78      public void notifyParameterEnd(String setName, String parameterName) {
79          System.out.println("End of parameter " + parameterName + " on "
80                  + setName);
81      }
82  
83  }
