20  package ch.powerunit.test.base;
21  
22  import java.util.Arrays;
23  import java.util.Collection;
24  import java.util.stream.Stream;
25  
26  import ch.powerunit.Categories;
27  import ch.powerunit.Parameter;
28  import ch.powerunit.Parameters;
29  import ch.powerunit.Test;
30  import ch.powerunit.TestSuite;
31  import ch.powerunit.impl.DefaultPowerUnitRunnerImpl;
32  
33  /**
34   * @author borettim
35   *
36   */
37  public class StreamParametersMapFunctionTests {
38  	public static void main(String[] args) {
39  		DefaultPowerUnitRunnerImpl<StreamParametersMapFunctionTest> runner = new DefaultPowerUnitRunnerImpl<>(
40  				StreamParametersMapFunctionTest.class);
41  		runner.addListener(new BootstrapTestListener<StreamParametersMapFunctionTest>());
42  		runner.run();
43  
44  	}
45  
46  	@Categories("base")
47  	public static class StreamParametersMapFunctionTest implements TestSuite {
48  		@Parameters
49  		public static Stream<Object[]> getDatas() {
50  			return Arrays
51  					.stream(new String[][] { { "a", "1", "java.lang.String",
52  							"3,4", "true,false" } })
53  					.map(DSL.stringToParameterMap(StreamParametersMapFunctionTest.class))
54  					.filter(DSL.parametersFilterUsingMatcher(DSL
55  							.arrayWithSize(5)));
56  		}
57  
58  		@Parameter(0)
59  		public String field1;
60  
61  		@Parameter(1)
62  		public int field2;
63  
64  		@Parameter(2)
65  		public Class<?> field3;
66  
67  		@Parameter(3)
68  		public String field4[];
69  
70  		@Parameter(4)
71  		public Collection<Boolean> field5;
72  
73  		@Test
74  		public void test() {
75  			assertThat(field1).is("a");
76  			assertThat(field2).is(1);
77  			assertThat(field3).is(equalTo(String.class));
78  			assertThat(field4).is(new String[] { "3", "4" });
79  			assertThatIterable(field5).contains(true, false);
80  		}
81  	}
82  }
