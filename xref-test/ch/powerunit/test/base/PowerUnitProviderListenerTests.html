20  package ch.powerunit.test.base;
21  
22  import org.apache.maven.surefire.report.CategorizedReportEntry;
23  import org.apache.maven.surefire.report.ConsoleLogger;
24  import org.apache.maven.surefire.report.ReportEntry;
25  import org.apache.maven.surefire.report.RunListener;
26  import org.apache.maven.surefire.report.SimpleReportEntry;
27  import org.mockito.Matchers;
28  import org.mockito.Mock;
29  import org.mockito.Mockito;
30  
31  import ch.powerunit.Categories;
32  import ch.powerunit.Rule;
33  import ch.powerunit.Test;
34  import ch.powerunit.TestContext;
35  import ch.powerunit.TestRule;
36  import ch.powerunit.TestSuite;
37  import ch.powerunit.impl.DefaultPowerUnitRunnerImpl;
38  import ch.powerunit.surefire.PowerUnitProviderListener;
39  
40  public class PowerUnitProviderListenerTests {
41  
42  	public static void main(String[] args) {
43  		DefaultPowerUnitRunnerImpl<PowerUnitProviderListenerTest> runner = new DefaultPowerUnitRunnerImpl<>(
44  				PowerUnitProviderListenerTest.class);
45  		runner.addListener(new BootstrapTestListener<PowerUnitProviderListenerTest>());
46  		runner.run();
47  	}
48  
49  	@Categories("base")
50  	public static class PowerUnitProviderListenerTest implements TestSuite {
51  		@Mock
52  		private RunListener rl;
53  
54  		private Class<?> underTest = Object.class;
55  
56  		@Mock
57  		private ConsoleLogger consoleLogger;
58  
59  		@Mock
60  		private TestContext<Object> textContext;
61  
62  		@Rule
63  		public final TestRule testRule = mockitoRule().around(
64  				before(this::generateSetup));
65  
66  		private PowerUnitProviderListener<Object> listener;
67  
68  		public void generateSetup() {
69  			listener = new PowerUnitProviderListener<>(consoleLogger, rl,
70  					underTest);
71  			Mockito.when(textContext.getFullTestName()).thenReturn("name");
72  			Mockito.when(textContext.getTestCategories()).thenReturn("parent");
73  
74  		}
75  
76  		@Test
77  		public void testNotifyStart() {
78  			listener.notifyStart(textContext);
79  			Mockito.verify(rl).testStarting(Matchers.any(ReportEntry.class));
80  		}
81  
82  		@Test
83  		public void testNotifySuccess() {
84  			listener.notifySuccess(textContext);
85  			Mockito.verify(rl).testSucceeded(Matchers.any(ReportEntry.class));
86  		}
87  
88  		@Test
89  		public void testNotifyFailure() {
90  			listener.notifyFailure(textContext, new Exception());
91  			Mockito.verify(rl).testFailed(Matchers.any(ReportEntry.class));
92  		}
93  
94  		@Test
95  		public void testNotifyError() {
96  			listener.notifyError(textContext, new Exception());
97  			Mockito.verify(rl).testError(Matchers.any(ReportEntry.class));
98  		}
99  
100 		@Test
101 		public void testNotifySetStartNotNull() {
102 			listener.notifySetStart("name", "parent");
103 			Mockito.verify(rl).testSetStarting(
104 					Matchers.any(CategorizedReportEntry.class));
105 		}
106 
107 		@Test
108 		public void testNotifySetStartNull() {
109 			listener.notifySetStart("name", null);
110 			Mockito.verify(rl).testSetStarting(
111 					Matchers.any(SimpleReportEntry.class));
112 		}
113 
114 		@Test
115 		public void testNotifySetEndNotNull() {
116 			listener.notifySetEnd("name", "parent");
117 			Mockito.verify(rl).testSetCompleted(
118 					Matchers.any(CategorizedReportEntry.class));
119 		}
120 
121 		@Test
122 		public void testNotifySetEndNull() {
123 			listener.notifySetEnd("name", null);
124 			Mockito.verify(rl).testSetCompleted(
125 					Matchers.any(SimpleReportEntry.class));
126 		}
127 
128 		@Test
129 		public void testNotifySkipped() {
130 			listener.notifySkipped(textContext);
131 			Mockito.verify(rl).testSkipped(Matchers.any(ReportEntry.class));
132 		}
133 	}
134 
135 }
