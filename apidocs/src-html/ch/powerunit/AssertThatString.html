020package ch.powerunit;
021
022import java.util.regex.Pattern;
023
024import org.hamcrest.Matchers;
025
026/**
027 * DSL for assertion on string result.
028 * <p>
029 * This interface is returned by the various methods
030 * {@link TestSuite#assertThat(String) assertThat} exposed by {@link TestSuite}.
031 *
032 * @author borettim
033 *
034 */
035public interface AssertThatString extends AssertThatObject<String> {
036        /**
037         * Check that a string another string.
038         * <p>
039         * <br>
040         * <i>By default, assertion can only be used from the main thread of the
041         * test ; When used from another thread, the assertion will be lost. In the
042         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
043         * annotation is used, the assertion may not be lost, in case the thread use
044         * an assertion method from the test object instance. </i>
045         * 
046         * @param substring
047         *            the other string.
048         * @return true if the assertion is valid ; If the assertion is false,
049         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
050         *         fail the test, else, return false and the test will be failed
051         *         later.
052         * @see Test#fastFail() The documentation of the <code>fastFail</code>
053         *      attribute of the <code>@Test</code> annotation, regarding the action
054         *      done by this assertion.
055         */
056        default boolean containsString(String substring) {
057                return is(Matchers.containsString(substring));
058        }
059
060        /**
061         * Check that a string starts with another one.
062         * <p>
063         * <br>
064         * <i>By default, assertion can only be used from the main thread of the
065         * test ; When used from another thread, the assertion will be lost. In the
066         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
067         * annotation is used, the assertion may not be lost, in case the thread use
068         * an assertion method from the test object instance. </i>
069         * 
070         * @param prefix
071         *            the prefix.
072         * @return true if the assertion is valid ; If the assertion is false,
073         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
074         *         fail the test, else, return false and the test will be failed
075         *         later.
076         * @see Test#fastFail() The documentation of the <code>fastFail</code>
077         *      attribute of the <code>@Test</code> annotation, regarding the action
078         *      done by this assertion.
079         */
080        default boolean startsWith(String prefix) {
081                return is(Matchers.startsWith(prefix));
082        }
083
084        /**
085         * Check that a string ends with another one.
086         * <p>
087         * <br>
088         * <i>By default, assertion can only be used from the main thread of the
089         * test ; When used from another thread, the assertion will be lost. In the
090         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
091         * annotation is used, the assertion may not be lost, in case the thread use
092         * an assertion method from the test object instance. </i>
093         * 
094         * @param prefix
095         *            the prefix.
096         * @return true if the assertion is valid ; If the assertion is false,
097         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
098         *         fail the test, else, return false and the test will be failed
099         *         later.
100         * @see Test#fastFail() The documentation of the <code>fastFail</code>
101         *      attribute of the <code>@Test</code> annotation, regarding the action
102         *      done by this assertion.
103         */
104        default boolean endsWith(String prefix) {
105                return is(Matchers.endsWith(prefix));
106        }
107
108        /**
109         * Validate a string with a {@link java.util.regex.Pattern}.
110         * <p>
111         * <br>
112         * <i>By default, assertion can only be used from the main thread of the
113         * test ; When used from another thread, the assertion will be lost. In the
114         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
115         * annotation is used, the assertion may not be lost, in case the thread use
116         * an assertion method from the test object instance. </i>
117         * 
118         * @param pattern
119         *            the pattern to be used.
120         * @return true if the assertion is valid ; If the assertion is false,
121         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
122         *         fail the test, else, return false and the test will be failed
123         *         later.
124         * @see Test#fastFail() The documentation of the <code>fastFail</code>
125         *      attribute of the <code>@Test</code> annotation, regarding the action
126         *      done by this assertion.
127         */
128        default boolean matchesRegex(Pattern pattern) {
129                return is(TestSuite.DSL.matchesRegex(pattern));
130        }
131
132        /**
133         * Validate a string with a regex.
134         * <p>
135         * <br>
136         * <i>By default, assertion can only be used from the main thread of the
137         * test ; When used from another thread, the assertion will be lost. In the
138         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
139         * annotation is used, the assertion may not be lost, in case the thread use
140         * an assertion method from the test object instance. </i>
141         * 
142         * 
143         * @param regex
144         *            The regex to be used for the validation.
145         * @return true if the assertion is valid ; If the assertion is false,
146         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
147         *         fail the test, else, return false and the test will be failed
148         *         later.
149         * @see Test#fastFail() The documentation of the <code>fastFail</code>
150         *      attribute of the <code>@Test</code> annotation, regarding the action
151         *      done by this assertion.
152         */
153        default boolean matchesRegex(String regex) {
154                return is(TestSuite.DSL.matchesRegex(regex));
155        }
156}
