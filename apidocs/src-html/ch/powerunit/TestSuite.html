041/**
042 * This is the interface to be implemented by test class, in order to have
043 * access to the test DSL.
044 * <p>
045 * This interface is only used to provide the DSL capabilities for the test ; It
046 * is not used as a marker by the framework.
047 * <p>
048 * Several functionalities are provided here :
049 * <ul>
050 * <li>Assertion capabilities (assertion on Object, Iterable, Function, Piece of
051 * code).</li>
052 * <li>Matchers to be used as end parameter of the assertion.</li>
053 * <li><code>{@link #before(Runnable...)}</code> to setup a before action.</li>
054 * <li><code>{@link #after(Runnable...)}</code> to setup a after action.</li>
055 * <li><code>{@link #mockitoRule()}</code> to setup mockito.</li>
056 * <li><code>{@link #systemPropertiesRule(String...)}</code> to restore
057 * properties system after test ; Several others method related to
058 * systemProperties exist.</li>
059 * <li><code>{@link #temporaryFolder()}</code> to support temporary folder.</li>
060 * <li><code>{@link #parametersMap(int, Function)}</code> to create Map function
061 * to be used on the stream to be provided by the {@link Parameters
062 * &#64;Parameters}. Other variants of this method may exist.</li>
063 * <li>Severals method of the form <code>tester...</code> (for instance
064 * <code>{@link #testerOfMatcher(Class)}</code>) that can be used with the
065 * <code>@{@link TestDelegate}</code> annotation to test some standard classes
066 * (like Matchers, Comparator, Function, etc).
067 * </ul>
068 *
069 * @author borettim
070 *
071 */
072public interface TestSuite extends Assert, Assume, Matchers,
073                TestFrameworkSupport {
074
075        /**
076         * A static field that is a testsuite (to avoid implementing TestSuite in
077         * test, in the rare case when it may be required).
078         * <p>
079         * The main use case is to access the stream functionnalities from a
080         * {@link Parameters &#64;Parameters} annotated method, has this method must
081         * be static.
082         */
083        static TestSuite DSL = new TestSuite() {
084        };
085
086        /**
087         * Build a before testrule.
088         * <p>
089         * The passed runnable will be used before each test. The exact location of
090         * the execution is depending on where this used on the testRule chain.
091         * <p>
092         * In the much simple case (just one method to be executed before each
093         * test), the syntax is :
094         * 
095         * <pre>
096         * &#064;Rule
097         * public TestRule rule = before(this::beforeMethodName);
098         * </pre>
099         * 
100         * @param befores
101         *            the befores
102         * @return {@link TestRule the rule chain}.
103         * @see Rule
104         */
105        default TestRule before(Runnable... befores) {
106                return Arrays.stream(befores).map(TestRule::before)
107                                .reduce((prev, next) -> prev.around(next)).get();
108        }
109
110        /**
111         * Build a after testrule.
112         * <p>
113         * The passed runnable will be used after each test. The exact location of
114         * the execution is depending on where this used on the testRule chain.
115         * <p>
116         * In the much simple case (just one method to be executed after each test),
117         * the syntax is :
118         * 
119         * <pre>
120         * &#064;Rule
121         * public TestRule rule = after(this::afterMethodName);
122         * </pre>
123         * 
124         * @param afters
125         *            the afters
126         * @return {@link TestRule the rule chain}.
127         * @see Rule
128         */
129        default TestRule after(Runnable... afters) {
130                return Arrays.stream(afters).map(TestRule::after)
131                                .reduce((prev, next) -> prev.around(next)).get();
132        }

146        /**
147         * Produces a new rule for the temporary folder.
148         * <p>
149         * As the {@link TemporaryFolder} rule provides several methods that are
150         * required for the test, except in the case when only this rule is
151         * required, a direct usage in the rule DSL is not adapted.
152         * 
153         * For instance, assuming that it is required to mix a before and the
154         * {@link TemporaryFolder} rule, the code will look like :
155         * 
156         * <pre>
157         * private TemporaryFolder temporary = temporaryFolder();
158         * 
159         * &#064;Rule
160         * public TestRule rule = before(this::beforeMethodName).around(temporary);
161         * </pre>
162         * 
163         * This is required to ensure that the method of the {@link TemporaryFolder}
164         * object can be used (using the field named <code>temporary</code>).
165         * 
166         * @return the temporary folder rule.
167         * @see Rule
168         */
169        default TemporaryFolder temporaryFolder() {
170                return temporaryFolderBuilder().build();
171        }

173        /**
174         * Produces a new rule builder for the temporary folder.
175         * <p>
176         * As the {@link TemporaryFolder} rule provides several methods that are
177         * required for the test, except in the case when only this rule is
178         * required, a direct usage in the rule DSL is not adapted.
179         * 
180         * For instance, assuming that it is required to mix a before and the
181         * {@link TemporaryFolder} rule, the code will look like :
182         * 
183         * <pre>
184         * private TemporaryFolder temporary = temporaryFolderBuilder().build();
185         * 
186         * &#064;Rule
187         * public TestRule rule = before(this::beforeMethodName).around(temporary);
188         * </pre>
189         * 
190         * This is required to ensure that the method of the {@link TemporaryFolder}
191         * object can be used (using the field named <code>temporary</code>).
192         * <p>
193         * The builder provide several capabilities to create initial folder
194         * structure at the same time than the temporary folder itself.
195         * 
196         * @return the temporary folder rule builder.
197         * @see Rule
198         */

200                return new TemporaryFolderImpl.TemporaryFolderBuilderImpl();
201        }

246        /**
247         * Start building a Parameter Mapper function, with an initial converter.
248         * <p>
249         * Not specified index are considered transformed by identity function.
250         * 
251         * @param idx
252         *            The parameter index
253         * @param mapFunction
254         *            the function to be applied
255         * @return the function on the parameter array
256         * @param <T>
257         *            The input type for the function
258         * @param <R>
259         *            the result type for the function
260         */
261        default <T, R> StreamParametersMapFunction<T> parametersMap(int idx,
262                        Function<T, R> mapFunction) {
263                return StreamParametersMapFunction.map(idx, mapFunction);
264        }

266        /**
267         * Start building a Parameter Mapper function, assuming that the input are
268         * String, and using the type of the {@link Parameter &#64;Parameter} field.
269         * <p>
270         * Fields not supported will not be mapped and must be handled manually,
271         * using {@link StreamParametersMapFunction#andMap(int, Function) andMap}
272         * method to avoid any unexpected error.
273         * <p>
274         * The goal of this method is to provide a way to receive so generic
275         * parameter and not having to care about typing. Let's take for example the
276         * following use case :
277         * <ul>
278         * <li>As an input for the test parameters, a CSV file is used. In this
279         * case, a framework like <a
280         * href="http://opencsv.sourceforge.net/">OpenCSV</a> can be used to load
281         * all the parameters, but they are all String.</li>
282         * <li>On the produced stream, the
283         * {@link java.util.stream.Stream#map(Function) map} method can be used to
284         * transform the received data into another format. Here, using the result
285         * of this method as parameter of the map method will ensure the
286         * transformation of the String to the right type, for simple type.</li>
287         * <li>For undetected field type, it is possible to use the method
288         * {@link StreamParametersMapFunction#andMap(int, Function) andMap} (on the
289         * returned object of this method), to add manual transformation.</li>
290         * </ul>
291         * <p>
292         * In this context, as the {@link Parameters &#64;Parameters} annotated
293         * method must be static, access to this method can be done using
294         * {@link #DSL DSL.} prefix.
295         * 
296         * @param testClass
297         *            the testClass, as this method is to be used in static mode.
298         * @return the function on the parameter array
299         * @see <a href="./helpers/doc-files/convertedType.html">Supported automated
300         *      conversion</a>
301         */
302        default StreamParametersMapFunction<String> stringToParameterMap(
303                        Class<?> testClass) {
304                return StreamParametersMapFunction.stringToParameterMap(testClass);
305        }

307        /**
308         * Provide a way to add a field to each parameter line.

310         * @param field

313         *         {@link java.util.stream.Stream#map(Function)}).
314         * @since 0.1.0
315         * @param <T>
316         *            The object type to be added.
317         */

319                return StreamParametersMapFunction.addFieldToEachEntry(field);
320        }

335        /**
336         * Expose a matcher as a predicate.
337         * 
338         * @param matcher
339         *            the matcher.
340         * @return the predicate
341         * @param <T>
342         *            The target object type
343         */
344        default <T> Predicate<T> matcherPredicate(Matcher<T> matcher) {
345                return matcher::matches;
346        }

449        /**
450         * Build a {@link TestListenerRule} based on the various method.
451         * 
452         * @param onStart
453         *            {@link TestListenerRule#onStart(TestContext) the action to be
454         *            done before the test start}. If null, nothing is done.
455         * @param onEnd
456         *            {@link TestListenerRule#onEnd(TestContext) the action to be
457         *            done after the test end}. If null, nothing is done.
458         * @param onFailure
459         *            {@link TestListenerRule#onFailure(TestContext, AssertionError)
460         *            the action to be done in case of failure}. If null, nothing is
461         *            done.
462         * @param onError
463         *            {@link TestListenerRule#onError(TestContext, Throwable) the
464         *            action to be done in case of error}. If null, nothing is done.
465         * @param onAssumptionSkip
466         *            {@link TestListenerRule#onAssumptionSkip(TestContext, AssumptionError)
467         *            the action to be done in case of assumption skipped}. If null
468         *            nothing is done.
469         * @return the Test Rule.
470         * @since 0.4.0
471         */
472        default TestListenerRule testListenerRuleOf(
473                        Consumer<TestContext<Object>> onStart,
474                        Consumer<TestContext<Object>> onEnd,
475                        BiConsumer<TestContext<Object>, AssertionError> onFailure,
476                        BiConsumer<TestContext<Object>, Throwable> onError,
477                        BiConsumer<TestContext<Object>, AssumptionError> onAssumptionSkip) {
478                return TestListenerRule.of(onStart, onEnd, onFailure, onError,
479                                onAssumptionSkip);
480        }
