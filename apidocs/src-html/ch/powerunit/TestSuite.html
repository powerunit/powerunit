039/**
040 * This is the interface to be implemented by test class, in order to have
041 * access to the test DSL.
042 * <p>
043 * This interface is only used to provide the DSL capabilities for the test ; It
044 * is not used as a marker by the framework.
045 * <p>
046 * Several functionalities are provided here :
047 * <ul>
048 * <li>Assertion capabilities (assertion on Object, Iterable, Function, Piece of
049 * code).</li>
050 * <li>Matchers to be used as end parameter of the assertion.</li>
051 * <li><code>{@link #before(Runnable...)}</code> to setup a before action.</li>
052 * <li><code>{@link #after(Runnable...)}</code> to setup a after action.</li>
053 * <li><code>{@link #mockitoRule()}</code> to setup mockito.</li>
054 * <li><code>{@link #systemPropertiesRule(String...)}</code> to restore
055 * properties system after test ; Several others method related to
056 * systemProperties exist.</li>
057 * <li><code>{@link #temporaryFolder()}</code> to support temporary folder.</li>
058 * <li><code>{@link #parametersMap(int, Function)}</code> to create Map function
059 * to be used on the stream to be provided by the {@link Parameters
060 * &#64;Parameters}. Other variants of this method may exist.</li>
061 * <li>Severals method of the form <code>tester...</code> (for instance
062 * <code>{@link #testerOfMatcher(Class)}</code>) that can be used with the
063 * <code>@{@link TestDelegate}</code> annotation to test some standard classes
064 * (like Matchers, Comparator, Function, etc).
065 * </ul>
066 *
067 * @author borettim
068 *
069 */
070public interface TestSuite extends Assert, Assume, Matchers,
071                TestFrameworkSupport {
072
073        /**
074         * A static field that is a testsuite (to avoid implementing TestSuite in
075         * test, in the rare case when it may be required).
076         * <p>
077         * The main use case is to access the stream functionnalities from a
078         * {@link Parameters &#64;Parameters} annotated method, has this method must
079         * be static.
080         */
081        static TestSuite DSL = new TestSuite() {
082        };
083
084        /**
085         * Build a before testrule.
086         * <p>
087         * The passed runnable will be used before each test. The exact location of
088         * the execution is depending on where this used on the testRule chain.
089         * <p>
090         * In the much simple case (just one method to be executed before each
091         * test), the syntax is :
092         * 
093         * <pre>
094         * &#064;Rule
095         * public TestRule rule = before(this::beforeMethodName);
096         * </pre>
097         * 
098         * @param befores
099         *            the befores
100         * @return {@link TestRule the rule chain}.
101         * @see Rule
102         */
103        default TestRule before(Runnable... befores) {
104                return Arrays.stream(befores).map(TestRule::before)
105                                .reduce((prev, next) -> prev.around(next)).get();
106        }
107
108        /**
109         * Build a after testrule.
110         * <p>
111         * The passed runnable will be used after each test. The exact location of
112         * the execution is depending on where this used on the testRule chain.
113         * <p>
114         * In the much simple case (just one method to be executed after each test),
115         * the syntax is :
116         * 
117         * <pre>
118         * &#064;Rule
119         * public TestRule rule = after(this::afterMethodName);
120         * </pre>
121         * 
122         * @param afters
123         *            the afters
124         * @return {@link TestRule the rule chain}.
125         * @see Rule
126         */
127        default TestRule after(Runnable... afters) {
128                return Arrays.stream(afters).map(TestRule::after)
129                                .reduce((prev, next) -> prev.around(next)).get();
130        }

136         *

138         * @see Rule

141                return TestRule.mockitoRule();
142        }

144        /**
145         * Produces a new rule for the temporary folder.
146         * <p>
147         * As the {@link TemporaryFolder} rule provides several methods that are
148         * required for the test, except in the case when only this rule is
149         * required, a direct usage in the rule DSL is not adapted.
150         * 
151         * For instance, assuming that it is required to mix a before and the
152         * {@link TemporaryFolder} rule, the code will look like :
153         * 
154         * <pre>
155         * private TemporaryFolder temporary = temporaryFolder();
156         * 
157         * &#064;Rule
158         * public TestRule rule = before(this::beforeMethodName).around(temporary);
159         * </pre>
160         * 
161         * This is required to ensure that the method of the {@link TemporaryFolder}
162         * object can be used (using the field named <code>temporary</code>).
163         * 
164         * @return the temporary folder rule.
165         * @see Rule
166         */
167        default TemporaryFolder temporaryFolder() {
168                return temporaryFolderBuilder().build();
169        }
170
171        /**
172         * Produces a new rule builder for the temporary folder.
173         * <p>
174         * As the {@link TemporaryFolder} rule provides several methods that are
175         * required for the test, except in the case when only this rule is
176         * required, a direct usage in the rule DSL is not adapted.
177         * 
178         * For instance, assuming that it is required to mix a before and the
179         * {@link TemporaryFolder} rule, the code will look like :
180         * 
181         * <pre>
182         * private TemporaryFolder temporary = temporaryFolderBuilder().build();
183         * 
184         * &#064;Rule
185         * public TestRule rule = before(this::beforeMethodName).around(temporary);
186         * </pre>
187         * 
188         * This is required to ensure that the method of the {@link TemporaryFolder}
189         * object can be used (using the field named <code>temporary</code>).
190         * <p>
191         * The builder provide several capabilities to create initial folder
192         * structure at the same time than the temporary folder itself.
193         * 
194         * @return the temporary folder rule builder.
195         * @see Rule
196         */
197        default TemporaryFolderBuilder temporaryFolderBuilder() {
198                return new TemporaryFolderImpl.TemporaryFolderBuilderImpl();
199        }

244        /**
245         * Start building a Parameter Mapper function, with an initial converter.
246         * <p>
247         * Not specified index are considered transformed by identity function.
248         * 
249         * @param idx
250         *            The parameter index
251         * @param mapFunction
252         *            the function to be applied
253         * @return the function on the parameter array
254         * @param <T>
255         *            The input type for the function
256         * @param <R>
257         *            the result type for the function
258         */
259        default <T, R> StreamParametersMapFunction<T> parametersMap(int idx,
260                        Function<T, R> mapFunction) {
261                return StreamParametersMapFunction.map(idx, mapFunction);
262        }
263
264        /**
265         * Start building a Parameter Mapper function, assuming that the input are
266         * String, and using the type of the {@link Parameter &#64;Parameter} field.
267         * <p>
268         * Fields not supported will not be mapped and must be handled manually,
269         * using {@link StreamParametersMapFunction#andMap(int, Function) andMap}
270         * method to avoid any unexpected error.
271         * <p>
272         * The goal of this method is to provide a way to receive so generic
273         * parameter and not having to care about typing. Let's take for example the
274         * following use case :
275         * <ul>
276         * <li>As an input for the test parameters, a CSV file is used. In this
277         * case, a framework like <a
278         * href="http://opencsv.sourceforge.net/">OpenCSV</a> can be used to load
279         * all the parameters, but they are all String.</li>
280         * <li>On the produced stream, the
281         * {@link java.util.stream.Stream#map(Function) map} method can be used to
282         * transform the received data into another format. Here, using the result
283         * of this method as parameter of the map method will ensure the
284         * transformation of the String to the right type, for simple type.</li>
285         * <li>For undetected field type, it is possible to use the method
286         * {@link StreamParametersMapFunction#andMap(int, Function) andMap} (on the
287         * returned object of this method), to add manual transformation.</li>
288         * </ul>
289         * <p>
290         * In this context, as the {@link Parameters &#64;Parameters} annotated
291         * method must be static, access to this method can be done using
292         * {@link #DSL DSL.} prefix.
293         * 
294         * @param testClass
295         *            the testClass, as this method is to be used in static mode.
296         * @return the function on the parameter array
297         * @see <a href="./helpers/doc-files/convertedType.html">Supported automated
298         *      conversion</a>
299         */
300        default StreamParametersMapFunction<String> stringToParameterMap(
301                        Class<?> testClass) {
302                return StreamParametersMapFunction.stringToParameterMap(testClass);
303        }
304
305        /**
306         * Provide a way to add a field to each parameter line.
307         * 
308         * @param field
309         *            The field to be added.
310         * @return the function that can be used on the stream (
311         *         {@link java.util.stream.Stream#map(Function)}).
312         * @since 0.1.0
313         * @param <T>
314         *            The object type to be added.
315         */
316        default <T> Function<Object[], Object[]> addFieldToEachEntry(T field) {
317                return StreamParametersMapFunction.addFieldToEachEntry(field);
318        }

320        /**
321         * Provide a filter for stream parameters, to keep only parameters accepted
322         * by a matcher.
323         * 
324         * @param matcher
325         *            the matcher
326         * @return the stream filter
327         */
328        default Predicate<Object[]> parametersFilterUsingMatcher(
329                        Matcher<Object[]> matcher) {
330                return matcherPredicate(matcher);
331        }
332
333        /**
334         * Expose a matcher as a predicate.
335         * 
336         * @param matcher
337         *            the matcher.
338         * @return the predicate
339         * @param <T>
340         *            The target object type
341         */
342        default <T> Predicate<T> matcherPredicate(Matcher<T> matcher) {
343                return matcher::matches;
344        }
