020package ch.powerunit.rules;
021
022import java.io.IOException;
023import java.nio.file.Path;
024import java.util.Collection;
025
026/**
027 * This rule provides a way to support temporary folder.
028 * <p>
029 * This class is exposed by the
030 * {@link ch.powerunit.TestSuite#temporaryFolder() temporaryFolder()} method of
031 * the {@link ch.powerunit.TestSuite TestSuite} interface. Created file and
032 * folder by this rule (or created inside the once created by this rule) are
033 * removed after test execution.
034 * 
035 * @author borettim
036 *
037 */
038public interface TemporaryFolder extends ExternalResource {
039
040        /**
041         * Create a new file with generated name.
042         * 
043         * @return the file
044         * @throws IOException
045         *             in case of error
046         */
047        Path newFile() throws IOException;
048
049        /**
050         * Create a new file with specified name.
051         * 
052         * @param fileName
053         *            the file name
054         * @return the file
055         * @throws IOException
056         *             in case of error
057         */
058        Path newFile(String fileName) throws IOException;
059
060        /**
061         * Create a new file with specified name and data.
062         * 
063         * @param fileName
064         *            the file name
065         * @param data
066         *            the byte that must be wrote into the file
067         * @return the file
068         * @throws IOException
069         *             in case of error
070         */
071        Path newFile(String fileName, byte data[]) throws IOException;
072
073        /**
074         * Create a new folder with generated name.
075         * 
076         * @return the folder
077         * @throws IOException
078         *             in case of error
079         */
080        Path newFolder() throws IOException;
081
082        /**
083         * Create a new folder with specified name.
084         * 
085         * @param folderName
086         *            the folder name
087         * @return the folder
088         * @throws IOException
089         *             in case of error
090         */
091        Path newFolder(String folderName) throws IOException;
092
093        /**
094         * Get the rootFolder.
095         * 
096         * @return the rootFolder
097         */
098        Path getRootFolder();
099
100        /**
101         * Get the initial temporary folder info
102         * 
103         * @return the initial data
104         */
105        InitialFolderEntry getInitial();
106
107        /**
108         * This is a builder for temporary folder.
109         * 
110         * @author borettim
111         *
112         */
113        interface TemporaryFolderBuilder {
114
115                /**
116                 * Create a new file, in the current folder.
117                 * 
118                 * @param fileName
119                 *            the file name
120                 * @return the temporary folder builder
121                 */
122                TemporaryFolderBuilder file(String fileName);
123
124                /**
125                 * Create a new file, in the current folder.
126                 * 
127                 * @param fileName
128                 *            the file name
129                 * @param data
130                 *            the data to be wrote in the file
131                 * @return the temporary folder builder
132                 */
133                TemporaryFolderBuilder file(String fileName, byte data[]);
134
135                /**
136                 * Create a new folder, in the current folder.
137                 * 
138                 * @param folderName
139                 *            the folder name
140                 * @return the temporary folder builder (moved in this folder)
141                 */
142                TemporaryFolderBuilder folder(String folderName);
143
144                /**
145                 * Move up from this folder.
146                 * 
147                 * @return the builder
148                 */
149                TemporaryFolderBuilder end();
150
151                /**
152                 * Build the temporary folder.
153                 * 
154                 * @return The temporary folder rule
155                 */
156                TemporaryFolder build();
157        }
158
159        /**
160         * Generic part regarding initial context of a temporary folder.
161         * 
162         * @author borettim
163         *
164         */
165        interface InitialEntry {
166                /**
167                 * Get the name (file or folder)
168                 * 
169                 * @return the name
170                 */
171                String getName();
172        }
173
174        /**
175         * An initial folder.
176         * 
177         * @author borettim
178         *
179         */
180        interface InitialFolderEntry extends InitialEntry {
181
182                /**
183                 * The list of files.
184                 * 
185                 * @return the list of files
186                 */
187                Collection<InitialFileEntry> getFiles();
188
189                /**
190                 * The list of folder.
191                 * 
192                 * @return the list of folders.
193                 */
194                Collection<InitialFolderEntry> getFolders();
195        }
196
197        /**
198         * An initial file.
199         * 
200         * @author borettim
201         *
202         */
203        interface InitialFileEntry extends InitialEntry {
204                /**
205                 * the initial data of the file
206                 * 
207                 * @return the data (can be null).
208                 */
209                byte[] getData();
210        }
211
212}
