020package ch.powerunit.rules;
021
022import java.util.Objects;
023import java.util.function.BiConsumer;
024import java.util.function.Consumer;
025
026import javax.jws.Oneway;
027
028import ch.powerunit.Statement;
029import ch.powerunit.TestContext;
030import ch.powerunit.TestRule;
031import ch.powerunit.exception.AssumptionError;
032
033/**
034 * This can be use to support taking action depending on the issue of the test.
035 * <p>
036 * The purpose is here to provide an interface that can be implemented by rule
037 * implementor to do action, before, after and on some condition.
038 * <p>
039 * Rule implementer should implements this interface and then implement the
040 * required methods (which are do-nothing method by default).
041 *
042 * The order of execution is the following :
043 * <ul>
044 * <li>Before the test, the method {@link #onStart(TestContext)} is executed.</li>
045 * <li>Then the test is executed.</li>
046 * <li>In case of error/failure, one (and only one) of the next methods is
047 * executed
048 * <ol>
049 * <li>{@link #onFailure(TestContext, AssertionError)} in case of test failure.</li>
050 * <li>{@link #onError(TestContext, Throwable)} in case of test error.</li>
051 * <li>{@link #onAssumptionSkip(TestContext, AssumptionError)} in case of test
052 * skip.</li>
053 * </ol>
054 * </li>
055 * <li>In all case, after the test, and after the previous method in case of
056 * error/failure, the method {@link #onEnd(TestContext)} is executed.</li>
057 * </ul>
058 *
059 * The {@link ExternalResource} rule implements this interface to provide simple
060 * use case (action before and always after test).
061 *
062 * @author borettim
063 * @see ExternalResource
064 */
065public interface TestListenerRule extends TestRule {
066
067        /**
068         * Method used at the start of the test.
069         * <p>
070         * Default implementation is to do nothing.
071         * 
072         * @param context
073         *            the test context
074         */
075        default void onStart(TestContext<Object> context) {
076                // Do nothing as default
077        }
078
079        /**
080         * Method used when a failure happened.
081         * <p>
082         * Default implementation is to do nothing.
083         * 
084         * @param context
085         *            the test context
086         * @param af
087         *            the failure
088         */
089        default void onFailure(TestContext<Object> context, AssertionError af) {
090                // Do nothing as default
091        }
092
093        /**
094         * Method used when an error happened.
095         * <p>
096         * Default implementation is to do nothing.
097         * 
098         * @param context
099         *            the test context
100         * @param error
101         *            the error
102         */
103        default void onError(TestContext<Object> context, Throwable error) {
104                // Do nothing as default
105        }
106
107        /**
108         * Method used when an assumption error happened.
109         * <p>
110         * Default implementation is to do nothing.
111         * 
112         * @param context
113         *            the test context
114         * @param error
115         *            the assumption error
116         */
117        default void onAssumptionSkip(TestContext<Object> context,
118                        AssumptionError error) {
119                // Do nothing as default
120        }
121
122        /**
123         * Method used at the end of the test.
124         * <p>
125         * Default implementation is to do nothing.
126         * 
127         * @param context
128         *            the test context
129         */
130        default void onEnd(TestContext<Object> context) {
131                // Do nothing as default
132        }
133
134        @Override
135        default Statement<TestContext<Object>, Throwable> computeStatement(
136                        Statement<TestContext<Object>, Throwable> inner) {
137                return (p) -> {
138                        try {
139                                onStart(p);
140                                inner.run(p);
141                        } catch (AssertionError af) {
142                                onFailure(p, af);
143                                throw af;
144                        } catch (InternalError ie) {
145                                onError(p, ie);
146                                throw ie;
147                        } catch (AssumptionError ie) {
148                                onAssumptionSkip(p, ie);
149                                throw ie;
150                        } catch (Throwable t) {
151                                onError(p, t);
152                                throw t;
153                        } finally {
154                                onEnd(p);
155                        }
156                };
157        }
158
159        /**
160         * Build a {@link TestListenerRule} based on the various method.
161         * 
162         * @param onStart
163         *            {@link #onStart(TestContext) the action to be done before the
164         *            test start}. If null, nothing is done.
165         * @param onEnd
166         *            {@link #onEnd(TestContext) the action to be done after the
167         *            test end}. If null, nothing is done.
168         * @param onFailure
169         *            {@link #onFailure(TestContext, AssertionError) the action to
170         *            be done in case of failure}. If null, nothing is done.
171         * @param onError
172         *            {@link #onError(TestContext, Throwable) the action to be done
173         *            in case of error}. If null, nothing is done.
174         * @param onAssumptionSkip
175         *            {@link #onAssumptionSkip(TestContext, AssumptionError) the
176         *            action to be done in case of assumption skipped}. If null
177         *            nothing is done.
178         * @return the Test Rule.
179         * @since 0.4.0
180         */
181        static TestListenerRule of(Consumer<TestContext<Object>> onStart,
182                        Consumer<TestContext<Object>> onEnd,
183                        BiConsumer<TestContext<Object>, AssertionError> onFailure,
184                        BiConsumer<TestContext<Object>, Throwable> onError,
185                        BiConsumer<TestContext<Object>, AssumptionError> onAssumptionSkip) {
186                return new TestListenerRule() {
187
188                        @Override
189                        public void onStart(TestContext<Object> context) {
190                                if (onStart != null) {
191                                        onStart.accept(context);
192                                }
193                        }
194
195                        @Override
196                        public void onFailure(TestContext<Object> context, AssertionError af) {
197                                if (onFailure != null) {
198                                        onFailure.accept(context, af);
199                                }
200                        }
201
202                        @Override
203                        public void onError(TestContext<Object> context, Throwable error) {
204                                if (onError != null) {
205                                        onError.accept(context, error);
206                                }
207                        }
208
209                        @Override
210                        public void onAssumptionSkip(TestContext<Object> context,
211                                        AssumptionError error) {
212                                if (onAssumptionSkip != null) {
213                                        onAssumptionSkip.accept(context, error);
214                                }
215                        }
216
217                        @Override
218                        public void onEnd(TestContext<Object> context) {
219                                if (onEnd != null) {
220                                        onEnd.accept(context);
221                                }
222                        }
223
224                };
225        }
226
227        /**
228         * Build a {@link TestListenerRule} with only an action at start.
229         * 
230         * @param onStart
231         *            {@link #onStart(TestContext) the action to be done before the
232         *            test start}.
233         * @return the Test Rule.
234         * @since 0.4.0
235         */
236        static TestListenerRule onStart(Consumer<TestContext<Object>> onStart) {
237                return of(Objects.requireNonNull(onStart, "onStart can't be null"),
238                                null, null, null, null);
239        }
240
241        /**
242         * Build a {@link TestListenerRule} with only an action at end.
243         * 
244         * @param onEnd
245         *            {@link #onEnd(TestContext) the action to be done after the
246         *            test end}.
247         * @return the Test Rule.
248         * @since 0.4.0
249         */
250        static TestListenerRule onEnd(Consumer<TestContext<Object>> onEnd) {
251                return of(null, Objects.requireNonNull(onEnd, "onEnd can't be null"),
252                                null, null, null);
253        }
254}
