020package ch.powerunit;
021
022import org.hamcrest.Matcher;
023import org.hamcrest.Matchers;
024
025/**
026 * DSL for assertion on object.
027 * <p>
028 * This interface is returned by the various methods
029 * {@link TestSuite#assertThat(Object) assertThat} exposed by {@link TestSuite}.
030 *
031 * @author borettim
032 * @param <T>
033 *            the object type.
034 */
035public interface AssertThatObject<T> {
036
037        /**
038         * Check that the object is matching the passed matcher.
039         * <p>
040         * <br>
041         * <i>By default, assertion can only be used from the main thread of the
042         * test ; When used from another thread, the assertion will be lost. In the
043         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
044         * annotation is used, the assertion may not be lost, in case the thread use
045         * an assertion method from the test object instance. </i>
046         * 
047         * @param matching
048         *            the matcher.
049         * @return true if the assertion is valid ; If the assertion is false,
050         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
051         *         fail the test, else, return false and the test will be failed
052         *         later.
053         * @see Test#fastFail() The documentation of the <code>fastFail</code>
054         *      attribute of the <code>@Test</code> annotation, regarding the action
055         *      done by this assertion.
056         */
057        boolean is(Matcher<? super T> matching);
058
059        /**
060         * Check that the object is the passed one (shortcut to Matcher.equalTo).
061         * <p>
062         * <br>
063         * <i>By default, assertion can only be used from the main thread of the
064         * test ; When used from another thread, the assertion will be lost. In the
065         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
066         * annotation is used, the assertion may not be lost, in case the thread use
067         * an assertion method from the test object instance. </i>
068         * 
069         * @param compareTo
070         *            the value.
071         * @return true if the assertion is valid ; If the assertion is false,
072         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
073         *         fail the test, else, return false and the test will be failed
074         *         later.
075         * @see Test#fastFail() The documentation of the <code>fastFail</code>
076         *      attribute of the <code>@Test</code> annotation, regarding the action
077         *      done by this assertion.
078         */
079        default boolean is(T compareTo) {
080                return is(Matchers.equalTo(compareTo));
081        }
082
083        /**
084         * Check that the object is not the passed one.
085         * <p>
086         * <br>
087         * <i>By default, assertion can only be used from the main thread of the
088         * test ; When used from another thread, the assertion will be lost. In the
089         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
090         * annotation is used, the assertion may not be lost, in case the thread use
091         * an assertion method from the test object instance. </i>
092         * 
093         * @param compareTo
094         *            the value
095         * @return true if the assertion is valid ; If the assertion is false,
096         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
097         *         fail the test, else, return false and the test will be failed
098         *         later.
099         * @see Test#fastFail() The documentation of the <code>fastFail</code>
100         *      attribute of the <code>@Test</code> annotation, regarding the action
101         *      done by this assertion.
102         */
103        default boolean isNot(T compareTo) {
104                return isNot(Matchers.equalTo(compareTo));
105        }
106
107        /**
108         * Check that the object is not matching the passed matcher.
109         * <p>
110         * <br>
111         * <i>By default, assertion can only be used from the main thread of the
112         * test ; When used from another thread, the assertion will be lost. In the
113         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
114         * annotation is used, the assertion may not be lost, in case the thread use
115         * an assertion method from the test object instance. </i>
116         * 
117         * @param matching
118         *            the matcher
119         * @return true if the assertion is valid ; If the assertion is false,
120         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
121         *         fail the test, else, return false and the test will be failed
122         *         later.
123         * @see Test#fastFail() The documentation of the <code>fastFail</code>
124         *      attribute of the <code>@Test</code> annotation, regarding the action
125         *      done by this assertion.
126         */
127        default boolean isNot(Matcher<T> matching) {
128                return is(Matchers.not(matching));
129        }
130
131        /**
132         * Check that the object is null.
133         * <p>
134         * <br>
135         * <i>By default, assertion can only be used from the main thread of the
136         * test ; When used from another thread, the assertion will be lost. In the
137         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
138         * annotation is used, the assertion may not be lost, in case the thread use
139         * an assertion method from the test object instance. </i>
140         * 
141         * @return true if the assertion is valid ; If the assertion is false,
142         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
143         *         fail the test, else, return false and the test will be failed
144         *         later.
145         * @see Test#fastFail() The documentation of the <code>fastFail</code>
146         *      attribute of the <code>@Test</code> annotation, regarding the action
147         *      done by this assertion.
148         */
149        default boolean isNull() {
150                return is(Matchers.nullValue());
151        }
152
153        /**
154         * Check that the object is not null.
155         * <p>
156         * <br>
157         * <i>By default, assertion can only be used from the main thread of the
158         * test ; When used from another thread, the assertion will be lost. In the
159         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
160         * annotation is used, the assertion may not be lost, in case the thread use
161         * an assertion method from the test object instance. </i>
162         * 
163         * @return true if the assertion is valid ; If the assertion is false,
164         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
165         *         fail the test, else, return false and the test will be failed
166         *         later.
167         * @see Test#fastFail() The documentation of the <code>fastFail</code>
168         *      attribute of the <code>@Test</code> annotation, regarding the action
169         *      done by this assertion.
170         */
171        default boolean isNotNull() {
172                return is(Matchers.notNullValue());
173        }
174
175        /**
176         * Check that the object is of a specific class (or subclasses).
177         * <p>
178         * <br>
179         * <i>By default, assertion can only be used from the main thread of the
180         * test ; When used from another thread, the assertion will be lost. In the
181         * case the {@link Test#fastFail() fastFail} attribute of {@link Test @Test}
182         * annotation is used, the assertion may not be lost, in case the thread use
183         * an assertion method from the test object instance. </i>
184         * 
185         * @param clazz
186         *            the class
187         * @return true if the assertion is valid ; If the assertion is false,
188         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
189         *         fail the test, else, return false and the test will be failed
190         *         later.
191         * @see Test#fastFail() The documentation of the <code>fastFail</code>
192         *      attribute of the <code>@Test</code> annotation, regarding the action
193         *      done by this assertion.
194         */
195        default boolean isA(Class<? super T> clazz) {
196                return is(Matchers.isA(clazz));
197        }
198
199}
