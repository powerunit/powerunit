020package ch.powerunit;
021
022import org.hamcrest.Matcher;
023import org.hamcrest.Matchers;
024
025/**
026 * DSL for assertion on object.
027 * <p>
028 * This interface is returned by the various methods
029 * {@link TestSuite#assertThat(Object) assertThat} exposed by {@link TestSuite}.
030 *
031 * @author borettim
032 * @param <T>
033 *            the object type.
034 */
035public interface AssertThatObject<T> {
036
037        /**
038         * Check that the object is matching the passed matcher.
039         * 
040         * @param matching
041         *            the matcher.
042         * @return true if the assertion is valid ; If the assertion is false,
043         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
044         *         fail the test, else, return false and the test will be failed
045         *         later.
046         */
047        boolean is(Matcher<? super T> matching);
048
049        /**
050         * Check that the object is the passed one (shortcut to Matcher.equalTo).
051         * 
052         * @param compareTo
053         *            the value.
054         * @return true if the assertion is valid ; If the assertion is false,
055         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
056         *         fail the test, else, return false and the test will be failed
057         *         later.
058         */
059        default boolean is(T compareTo) {
060                return is(Matchers.equalTo(compareTo));
061        }
062
063        /**
064         * Check that the object is not the passed one.
065         * 
066         * @param compareTo
067         *            the value
068         * @return true if the assertion is valid ; If the assertion is false,
069         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
070         *         fail the test, else, return false and the test will be failed
071         *         later.
072         */
073        default boolean isNot(T compareTo) {
074                return isNot(Matchers.equalTo(compareTo));
075        }
076
077        /**
078         * Check that the object is not matching the passed matcher.
079         * 
080         * @param matching
081         *            the matcher
082         * @return true if the assertion is valid ; If the assertion is false,
083         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
084         *         fail the test, else, return false and the test will be failed
085         *         later.
086         */
087        default boolean isNot(Matcher<T> matching) {
088                return is(Matchers.not(matching));
089        }
090
091        /**
092         * Check that the object is null.
093         * 
094         * @return true if the assertion is valid ; If the assertion is false,
095         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
096         *         fail the test, else, return false and the test will be failed
097         *         later.
098         */
099        default boolean isNull() {
100                return is(Matchers.nullValue());
101        }
102
103        /**
104         * Check that the object is not null.
105         * 
106         * @return true if the assertion is valid ; If the assertion is false,
107         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
108         *         fail the test, else, return false and the test will be failed
109         *         later.
110         */
111        default boolean isNotNull() {
112                return is(Matchers.notNullValue());
113        }
114
115        /**
116         * Check that the object is of a specific class (or subclasses).
117         * 
118         * @param clazz
119         *            the class
120         * @return true if the assertion is valid ; If the assertion is false,
121         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
122         *         fail the test, else, return false and the test will be failed
123         *         later.
124         */
125        default boolean isA(Class<? super T> clazz) {
126                return is(Matchers.isA(clazz));
127        }
128
129}
