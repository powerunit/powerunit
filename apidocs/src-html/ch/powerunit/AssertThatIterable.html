020package ch.powerunit;
021
022import org.hamcrest.Matcher;
023import org.hamcrest.Matchers;
024
025/**
026 * DSL on assertion on iterable.
027 * <p>
028 * This interface is returned by the various methods
029 * {@link TestSuite#assertThatIterable(Iterable) assertThatIterable} exposed by
030 * {@link TestSuite}.
031 *
032 * @author borettim
033 * @param <T>
034 *            the Element type of the Iterator
035 * @see AssertThatObject assertion on object are also available on iterable.
036 */
037public interface AssertThatIterable<T> extends AssertThatObject<Iterable<T>> {
038        /**
039         * Check the size of the iterable.
040         * 
041         * @param size
042         *            the expected size.
043         * @return true if the assertion is valid ; If the assertion is false,
044         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
045         *         fail the test, else, return false and the test will be failed
046         *         later.
047         */
048        @SuppressWarnings({ "unchecked", "rawtypes" })
049        default boolean hasSize(int size) {
050                return is((Matcher) Matchers.hasSize(size));
051        }
052
053        /**
054         * Check the size is not the one passed.
055         * 
056         * @param size
057         *            the not expected size.
058         * @return true if the assertion is valid ; If the assertion is false,
059         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
060         *         fail the test, else, return false and the test will be failed
061         *         later.
062         */
063        @SuppressWarnings({ "unchecked", "rawtypes" })
064        default boolean hasNotSize(int size) {
065                return isNot((Matcher) Matchers.hasSize(size));
066        }
067
068        /**
069         * Check that the iterable contains the items.
070         * 
071         * @param items
072         *            the expected items.
073         * @return true if the assertion is valid ; If the assertion is false,
074         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
075         *         fail the test, else, return false and the test will be failed
076         *         later.
077         */
078        default boolean contains(@SuppressWarnings("unchecked") T... items) {
079                return is(Matchers.contains(items));
080        }
081
082        /**
083         * Check that the iterable contains the items.
084         * 
085         * @param items
086         *            the matcher for each item.
087         * @return true if the assertion is valid ; If the assertion is false,
088         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
089         *         fail the test, else, return false and the test will be failed
090         *         later.
091         */
092        default boolean containsMatching(
093                        @SuppressWarnings("unchecked") Matcher<? super T>... items) {
094                return is(Matchers.contains(items));
095        }
096
097        /**
098         * Check that the iterable contains the items in any order.
099         * 
100         * @param items
101         *            the expected items.
102         * @return true if the assertion is valid ; If the assertion is false,
103         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
104         *         fail the test, else, return false and the test will be failed
105         *         later.
106         */
107        default boolean containsInAnyOrder(
108                        @SuppressWarnings("unchecked") T... items) {
109                return is(Matchers.containsInAnyOrder(items));
110        }
111
112        /**
113         * Check that the iterable contains the items in any order.
114         * 
115         * @param items
116         *            the matcher for each item.
117         * @return true if the assertion is valid ; If the assertion is false,
118         *         depending on {@link Test#fastFail()} : If <code>true</code>, then
119         *         fail the test, else, return false and the test will be failed
120         *         later.
121         */
122        default boolean containsInAnyOrderMatching(
123                        @SuppressWarnings("unchecked") Matcher<? super T>... items) {
124                return is(Matchers.containsInAnyOrder(items));
125        }
126}
