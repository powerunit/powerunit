020package org.powerunit;
021
022import java.util.Arrays;
023import java.util.function.Function;
024import java.util.function.Predicate;
025import java.util.function.Supplier;
026
027import org.hamcrest.Matcher;
028import org.powerunit.helpers.StreamParametersMapFunction;
029import org.powerunit.rules.SystemPropertiesRule;
030import org.powerunit.rules.TemporaryFolder;
031
032/**
033 * This is the interface to be implemented by test class, in order to have
034 * access to the test DSL.
035 * <p>
036 * This interface is only used to provide the DSL capabilities for the test ; It
037 * is not used as a marker by the framework.
038 * <p>
039 * Several functionalities are provided here :
040 * <ul>
041 * <li>Assertion capabilities (assertion on Object, Iterable, Function, Piece of
042 * code).</li>
043 * <li>Matchers to be used as end parameter of the assertion.</li>
044 * <li><code>{@link #before(Runnable...)}</code> to setup a before action.</li>
045 * <li><code>{@link #after(Runnable...)}</code> to setup a after action.</li>
046 * <li><code>{@link #mockitoRule()}</code> to setup mockito.</li>
047 * <li><code>{@link #systemPropertiesRule(String...)}</code> to restore
048 * properties system after test ; Several others method related to
049 * systemProperties exist.</li>
050 * <li><code>{@link #temporaryFolder()}</code> to support temporary folder.</li>
051 * <li><code>{@link #parametersMap(int, Function)}</code> to create Map function
052 * to be used on the stream to be provided by the {@link Parameters
053 * &#64;Parameters}. Other variants of this method may exist.</li>
054 * </ul>
055 * 
056 * @author borettim
057 *
058 */
059public interface TestSuite extends Assert, Assume, Matchers {
060
061        /**
062         * A static field that is a testsuite (to avoid implementing TestSuite in
063         * test, in the rare case when it may be required).
064         * <p>
065         * The main use case is to access the stream functionnalities from a
066         * {@link Parameters &#64;Parameters} annotated method, has this method must
067         * be static.
068         */
069        static TestSuite DSL = new TestSuite() {
070        };
071
072        /**
073         * Build a before testrule.
074         * <p>
075         * The passed runnable will be used before each test. The exact location of
076         * the execution is depending on where this used on the testRule chain.
077         * <p>
078         * In the much simple case (just one method to be executed before each
079         * test), the syntax is :
080         * 
081         * <pre>
082         * &#064;Rule
083         * public TestRule rule = before(this::beforeMethodName);
084         * </pre>
085         * 
086         * @param befores
087         *            the befores
088         * @return {@link TestRule the rule chain}.
089         * @see Rule
090         */
091        default TestRule before(Runnable... befores) {
092                return Arrays.stream(befores).map(TestRule::before)
093                                .reduce((prev, next) -> prev.around(next)).get();
094        }
095
096        /**
097         * Build a after testrule.
098         * <p>
099         * The passed runnable will be used after each test. The exact location of
100         * the execution is depending on where this used on the testRule chain.
101         * <p>
102         * In the much simple case (just one method to be executed after each test),
103         * the syntax is :
104         * 
105         * <pre>
106         * &#064;Rule
107         * public TestRule rule = after(this::afterMethodName);
108         * </pre>
109         * 
110         * @param afters
111         *            the afters
112         * @return {@link TestRule the rule chain}.
113         * @see Rule
114         */
115        default TestRule after(Runnable... afters) {
116                return Arrays.stream(afters).map(TestRule::after)
117                                .reduce((prev, next) -> prev.around(next)).get();
118        }
119
120        /**
121         * Create a rule to support mockito.
122         * <p>
123         * This provide a way to setup Mockito before each test.
124         * 
125         * @return {@link TestRule the rule chain}.
126         * @see Rule
127         */
128        default TestRule mockitoRule() {
129                return TestRule.mockitoRule();
130        }
131
132        /**
133         * Produces a new rule for the temporary folder.
134         * <p>
135         * As the {@link TemporaryFolder} rule provides several methods that are
136         * required for the test, except in the case when only this rule is
137         * required, a direct usage in the rule DSL is not adapted.
138         * 
139         * For instance, assuming that it is required to mix a before and the
140         * {@link TemporaryFolder} rule, the code will look like :
141         * 
142         * <pre>
143         * private TemporaryFolder temporary = temporaryFolder();
144         * 
145         * &#64;
146         * public TestRule rule = before(this::beforeMethodName).around(temporary);
147         * </pre>
148         * 
149         * This is required to ensure that the method of the {@link TemporaryFolder}
150         * object can be used (using the field named <code>temporary</code>).
151         * 
152         * @return the temporary folder rule.
153         * @see Rule
154         */
155        default TemporaryFolder temporaryFolder() {
156                return new TemporaryFolder();
157        }
158
159        /**
160         * Create a rule to restore some system properties after the test
161         * 
162         * @param propertiesName
163         *            the properties to be restored
164         * @return {@link TestRule the rule chain}.
165         * @see Rule
166         */
167        default TestRule systemPropertiesRule(String... propertiesName) {
168                return new SystemPropertiesRule(propertiesName);
169        }
170
171        /**
172         * Set a property before the run and ensure correct restore.
173         * 
174         * @param propertyName
175         *            the name of the property
176         * @param propertyValue
177         *            the value of the property
178         * @return {@link TestRule the rule chain}.
179         * @see Rule
180         */
181        default TestRule systemProperty(String propertyName,
182                        Supplier<String> propertyValue) {
183                return SystemPropertiesRule.setSystemPropertyBeforeTestAndRestoreAfter(
184                                propertyName, propertyValue);
185        }
186
187        /**
188         * Set a property before the run and ensure correct restore.
189         * 
190         * @param propertyName
191         *            the name of the property
192         * @param propertyValue
193         *            the value of the property
194         * @return {@link TestRule the rule chain}.
195         * @see Rule
196         */
197        default TestRule systemProperty(String propertyName, String propertyValue) {
198                return SystemPropertiesRule.setSystemPropertyBeforeTestAndRestoreAfter(
199                                propertyName, propertyValue);
200        }
201
202        /**
203         * Start building a Parameter Mapper function, with an initial converter.
204         * <p>
205         * Not specified index are considered transformed by identity function.
206         * 
207         * @param idx
208         *            The parameter index
209         * @param mapFunction
210         *            the function to be applied
211         * @return the function on the parameter array
212         * @param <T>
213         *            The input type for the function
214         * @param <R>
215         *            the result type for the function
216         */
217        default <T, R> StreamParametersMapFunction<T> parametersMap(int idx,
218                        Function<T, R> mapFunction) {
219                return StreamParametersMapFunction.map(idx, mapFunction);
220        }
221
222        /**
223         * Start building a Parameter Mapper function, assuming that the input are
224         * String, and using the type of the {@link Parameter &#64;Parameter} field.
225         * <p>
226         * Fields not supported will not be mapped and must be handled manually,
227         * using {@link StreamParametersMapFunction#andMap(int, Function) andMap}
228         * method to avoid any unexpected error.
229         * <p>
230         * The goal of this method is to provide a way to receive so generic
231         * parameter and not having to care about typing. Let's take for example the
232         * following use case :
233         * <ul>
234         * <li>As an input for the test parameters, a CSV file is used. In this
235         * case, a framework like <a
236         * href="http://opencsv.sourceforge.net/">OpenCSV</a> can be used to load
237         * all the parameters, but they are all String.</li>
238         * <li>On the produced stream, the
239         * {@link java.util.stream.Stream#map(Function) map} method can be used to
240         * transform the received data into another format. Here, using the result
241         * of this method as parameter of the map method will ensure the
242         * transformation of the String to the right type, for simple type.</li>
243         * <li>For undetected field type, it is possible to use the method
244         * {@link StreamParametersMapFunction#andMap(int, Function) andMap} (on the
245         * returned object of this method), to add manual transformation.</li>
246         * </ul>
247         * <p>
248         * In this context, as the {@link Parameters &#64;Parameters} annotated
249         * method must be static, access to this method can be done using
250         * {@link #DSL DSL.} prefix.
251         * 
252         * @param testClass
253         *            the testClass, as this method is to be used in static mode.
254         * @return the function on the parameter array
255         * 
256         */
257        default StreamParametersMapFunction<String> stringToParameterMap(
258                        Class<?> testClass) {
259                return StreamParametersMapFunction.stringToParameterMap(testClass);
260        }
261
262        /**
263         * Provide a filter for stream parameters, to keep only parameters accepted
264         * by a matcher.
265         * 
266         * @param matcher
267         *            the matcher
268         * @return the stream filter
269         */
270        default Predicate<Object[]> parametersFilterUsingMatcher(
271                        Matcher<Object[]> matcher) {
272                return matcherPredicate(matcher);
273        }
274
275        /**
276         * Expose a matcher as a predicate.
277         * 
278         * @param matcher
279         *            the matcher.
280         * @return the predicate
281         * @param <T>
282         *            The target object type
283         */
284        default <T> Predicate<T> matcherPredicate(Matcher<T> matcher) {
285                return matcher::matches;
286        }
287}
