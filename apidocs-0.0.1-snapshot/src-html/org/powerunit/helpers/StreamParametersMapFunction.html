020package org.powerunit.helpers;
021
022import java.util.Arrays;
023import java.util.HashMap;
024import java.util.Map;
025import java.util.Objects;
026import java.util.function.Function;
027
028import org.powerunit.Parameter;
029
030/**
031 * This is a class that provide a way to transform stream of String to Object
032 * based on the @Parameter field type
033 * 
034 * @author borettim
035 * @param <T>
036 *            the input type
037 */
038public final class StreamParametersMapFunction<T> implements
039                Function<T[], Object[]> {
040
041        private StreamParametersMapFunction() {
042        }
043
044        private Map<Integer, Function<Object, Object>> mapper = new HashMap<>();
045
046        @Override
047        public Object[] apply(T[] input) {
048                Object output[] = new Object[input.length];
049                for (int i = 0; i < input.length; i++) {
050                        Function<Object, Object> f = mapper.getOrDefault(i,
051                                        Function.identity());
052                        output[i] = f.apply(input[i]);
053                }
054                return output;
055        }
056
057        /**
058         * Start building a Parameter Mapper function, with an initial converter.
059         * <p>
060         * Not specified index are considered transformed by identity function.
061         * 
062         * @param idx
063         *            The parameter index
064         * @param mapFunction
065         *            the function to be applied
066         * @return the function on the parameter array
067         * @param <T>
068         *            The input type for the function
069         * @param <R>
070         *            the result type for the function
071         */
072        public static <T, R> StreamParametersMapFunction<T> map(int idx,
073                        Function<T, R> mapFunction) {
074                if (idx < 0) {
075                        throw new IllegalArgumentException("idx can't be negative");
076                }
077                return new StreamParametersMapFunction<T>().andMap(idx, mapFunction);
078        }
079
080        /**
081         * Start building a Parameter Mapper function, assuming that the input are
082         * String, and using the type of the {@link Parameter &#64;Parameter} field.
083         * <p>
084         * Fields not supported will not be mapped and must be handled manually,
085         * using {@link StreamParametersMapFunction#andMap(int, Function) andMap}
086         * method.
087         * 
088         * @param testClass
089         *            the testClass with the annotation
090         * @return the function on the parameter array
091         */
092        public static StreamParametersMapFunction<String> stringToParameterMap(
093                        Class<?> testClass) {
094                StreamParametersMapFunction<String> map = new StreamParametersMapFunction<>();
095                Arrays.stream(testClass.getDeclaredFields())
096                                .filter(f -> f.isAnnotationPresent(Parameter.class))
097                                .forEach(
098                                                f -> {
099                                                        if (f.getGenericType() instanceof Class) {
100                                                                int pid = f.getAnnotation(Parameter.class)
101                                                                                .value();
102                                                                Class<?> c = (Class<?>) f.getGenericType();
103                                                                if (int.class.equals(c)
104                                                                                || Integer.class.equals(c)) {
105                                                                        map.<Integer> andMap(pid, Integer::valueOf);
106                                                                } else if (float.class.equals(c)
107                                                                                || Float.class.equals(c)) {
108                                                                        map.<Float> andMap(pid, Float::valueOf);
109                                                                } else if (short.class.equals(c)
110                                                                                || Short.class.equals(c)) {
111                                                                        map.<Short> andMap(pid, Short::valueOf);
112                                                                } else if (double.class.equals(c)
113                                                                                || Double.class.equals(c)) {
114                                                                        map.<Double> andMap(pid, Double::valueOf);
115                                                                } else if (long.class.equals(c)
116                                                                                || Long.class.equals(c)) {
117                                                                        map.<Long> andMap(pid, Long::valueOf);
118                                                                } else if (char.class.equals(c)
119                                                                                || Character.class.equals(c)) {
120                                                                        map.<Character> andMap(pid,
121                                                                                        s -> s.charAt(0));
122                                                                } else if (String.class.equals(c)) {
123                                                                        map.<String> andMap(pid,
124                                                                                        Function.identity());
125                                                                } else if (boolean.class.equals(c)
126                                                                                || Boolean.class.equals(c)) {
127                                                                        map.<Boolean> andMap(pid, Boolean::valueOf);
128                                                                } else if (Class.class.equals(c)) {
129                                                                        map.<Class<?>> andMap(
130                                                                                        pid,
131                                                                                        s -> {
132                                                                                                try {
133                                                                                                        return Class.forName(s);
134                                                                                                } catch (ClassNotFoundException e) {
135                                                                                                        throw new IllegalArgumentException(
136                                                                                                                        "Unexpected error "
137                                                                                                                                        + e.getMessage(),
138                                                                                                                        e);
139                                                                                                }
140                                                                                        });
141                                                                }
142                                                        }
143                                                });
144                return map;
145        }
146
147        /**
148         * Defines an additional Parameter Mapper function.
149         * <p>
150         * Not specified index are considered transformed by identity function.
151         * 
152         * @param idx
153         *            The parameter index
154         * @param mapFunction
155         *            the function to be applied
156         * @return the function on the parameter array
157         * @param <R>
158         *            The return type for the function
159         */
160        public <R> StreamParametersMapFunction<T> andMap(int idx,
161                        Function<T, R> mapFunction) {
162                if (idx < 0) {
163                        throw new IllegalArgumentException("idx can't be negative");
164                }
165                Objects.requireNonNull(mapFunction);
166                mapper.put(idx, (Function) mapFunction);
167                return this;
168        }
169}
